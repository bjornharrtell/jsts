  4 /**
  5  * A Quadtree is a spatial index structure for efficient querying of 2D
  6  * rectangles. If other kinds of spatial objects need to be indexed they can be
  7  * represented by their envelopes
  8  * <p>
  9  * The quadtree structure is used to provide a primary filter for range
 10  * rectangle queries. The query() method returns a list of all objects which
 11  * <i>may</i> intersect the query rectangle. Note that it may return objects
 12  * which do not in fact intersect. A secondary filter is required to test for
 13  * exact intersection. Of course, this secondary filter may consist of other
 14  * tests besides intersection, such as testing other kinds of spatial
 15  * relationships.
 16  *
 17  * <p>
 18  * This implementation does not require specifying the extent of the inserted
 19  * items beforehand. It will automatically expand to accomodate any extent of
 20  * dataset.
 21  * <p>
 22  * This data structure is also known as an <i>MX-CIF quadtree</i> following the
 23  * usage of Samet and others.
 24  *
 25  * @constructor
 26  */
 27 jsts.index.quadtree.Quadtree = OpenLayers.Class();
 28 
 29 /**
 30  * Ensure that the envelope for the inserted item has non-zero extents. Use the
 31  * current minExtent to pad the envelope, if necessary
 32  */
 33 
 34 
 35 /**
 36  * Ensures an extent is not zero.
 37  *
 38  * @param {jsts.geom.Envelope}
 39  *          itemEnv The envelope to check.
 40  * @param {Number}
 41  *          minExtent the minimum width/height to expand the extent with if it
 42  *          is zero.
 43  * @return {jsts.geom.Envelope}
 44  *          A valid extent.
 45  */
 46 jsts.index.quadtree.Quadtree.prototype.ensureExtent = function(itemEnv,
 47     minExtent) {
 48   var minx, maxx, miny, maxy;
 49 
 50   minx = itemEnv.getMinX();
 51   maxx = itemEnv.getMaxX();
 52   miny = itemEnv.getMinY();
 53   maxy = itemEnv.getMaxY();
 54 
 55   // has a non-zero extent
 56   if (minx !== maxx && miny !== maxy) {
 57     return itemEnv;
 58   }
 59 
 60   // pad one or both extents
 61   if (minx === maxx) {
 62     minx = minx - (minExtent / 2.0);
 63     maxx = minx + (minExtent / 2.0);
 64   }
 65 
 66   if (miny === maxy) {
 67     miny = miny - (minExtent / 2.0);
 68     maxy = miny + (minExtent / 2.0);
 69   }
 70 
 71   return new jsts.geom.Envelope(minx, maxx, miny, maxy);
 72 };

 75 /**
 76  * Initializes a new quadtree using the OpenLayers inheritance mechanism
 77  */
 78 jsts.index.quadtree.Quadtree.prototype.initialize = function() {
 79   this.root = new jsts.index.quadtree.Root();
 80 
 81   /**
 82    * minExtent is the minimum envelope extent of all items inserted into the
 83    * tree so far. It is used as a heuristic value to construct non-zero
 84    * envelopes for features with zero X and/or Y extent. Start with a non-zero
 85    * extent, in case the first feature inserted has a zero extent in both
 86    * directions. This value may be non-optimal, but only one feature will be
 87    * inserted with this value.
 88    */
 89   this.minExtent = 1.0;
 90 };

115 /**
116  * Inserts an item to the tree
117  *
118  * @param {jsts.geom.Envelope}
119  *        itemEnv The envelope.
120  * @param {Object}
121  *        item The item.
122  */
123 jsts.index.quadtree.Quadtree.prototype.insert = function(itemEnv, item) {
124   this.collectStats(itemEnv);
125   var insertEnv = this.ensureExtent(itemEnv, this.minExtent);
126   this.root.insert(insertEnv, item);
127 };

130 /**
131  * Removes a single item from the tree
132  *
133  * @param {jsts.geom.Envelope}
134  *        itemEnv the envelope of the item to be removed.
135  * @param {Object}
136  *        item the item to remove.
137  * @return {Boolean}
138  *         <code>true</true> if the item was found (and removed).
139  */
140 jsts.index.quadtree.Quadtree.prototype.remove = function(itemEnv, item) {
141   var posEnv = this.ensureExtent(itemEnv, this.minExtent);
142   return this.root.remove(posEnv, item);
143 };

146 /**
147  * Querys the quadtree.
148  *
149  * Calls appropriate function depending on arguments
150  */
151 jsts.index.quadtree.Quadtree.prototype.query = function() {
152   if (arguments.length === 1) {
153     return jsts.index.quadtree.Quadtree.prototype.queryByEnvelope.apply(this,
154         arguments);
155   } else {
156     jsts.index.quadtree.Quadtree.prototype.queryWithVisitor.apply(this,
157         arguments);
158   }
159 };

162 /**
163  * Queries the tree and returns items which may lie in the given search
164  * envelope. Precisely, the items that are returned are all items in the tree
165  * whose envelope <b>may</b> intersect the search Envelope. Note that some
166  * items with non-intersecting envelopes may be returned as well; the client is
167  * responsible for filtering these out. In most situations there will be many
168  * items in the tree which do not intersect the search envelope and which are
169  * not returned - thus providing improved performance over a simple linear scan.
170  *
171  * @param {jsts.geom.Envelope}
172  *        searchEnv the envelope of the desired query area.
173  * @return {Array}
174  *         an array of items which may intersect the search envelope.
175  */
176 jsts.index.quadtree.Quadtree.prototype.queryByEnvelope = function(searchEnv) {
177   var visitor = new jsts.index.ArrayListVisitor();
178   this.query(searchEnv, visitor);
179 
180   return visitor.getItems();
181 };
182

184 /**
185  * Queries the tree and visits items which may lie in the given search envelope.
186  * Precisely, the items that are visited are all items in the tree whose
187  * envelope <b>may</b> intersect the search Envelope. Note that some items with
188  * non-intersecting envelopes may be visited as well; the client is responsible
189  * for filtering these out. In most situations there will be many items in the
190  * tree which do not intersect the search envelope and which are not visited -
191  * thus providing improved performance over a simple linear scan.
192  *
193  * @param {jsts.geom.Envelope}
194  *        searchEnv the envelope of the desired query area.
195  * @param {jsts.index.Visitor}
196  *        visitor a visitor object which is passed the visited items.
197  */
198 jsts.index.quadtree.Quadtree.prototype.queryWithVisitor = function(searchEnv,
199     visitor) {
200   this.root.visit(searchEnv, visitor);
201 };

204 /**
205  * Returns an array of all items in the quadtree.
206  *
207  * @return {Array}
208  *         An array of all items in the quadtree.
209  */
210 jsts.index.quadtree.Quadtree.prototype.queryAll = function() {
211   var foundItems = [];
212   foundItems = this.root.addAllItems(foundItems);
213   return foundItems;

217 /**
218  * Checks wheter a width and height of an envelope is above zero and sets
219  * minExtent if the widht or height is less than the current min extent
220  *
221  * @param {jsts.geom.Envelope}
222  *        itemEnv The envelope.
223  */
224 jsts.index.quadtree.Quadtree.prototype.collectStats = function(itemEnv) {
225   var delX = itemEnv.getWidth();
226   if (delX < this.minExtent && delX > 0.0) {
227     this.minExtent = delX;
228   }
229 
230   var delY = itemEnv.getHeight();
231   if (delY < this.minExtent && delY > 0.0) {
232     this.minExtent = delY;
233   }
234 };
