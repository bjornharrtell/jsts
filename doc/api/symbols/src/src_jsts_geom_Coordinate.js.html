  7 /**
  8  * A lightweight class used to store coordinates on the 2-dimensional Cartesian
  9  * plane. It is distinct from {@link Point}, which is a subclass of
 10  * {@link Geometry}. Unlike objects of type {@link Point} (which contain
 11  * additional information such as an envelope, a precision model, and spatial
 12  * reference system information), a <code>Coordinate</code> only contains
 13  * coordinate values and accessor methods.
 14  *
 15  * @requires jsts/geom/Geometry.js
 16  */
 17 
 18 
 19

 21  * @constructor

 24  */

 53 /**
 54  * Computes the 2-dimensional Euclidean distance to another location. The
 55  * Z-ordinate is ignored.
 56  *
 57  * @param {Coordinate}
 58  *          p a point.
 59  * @return {double} the 2-dimensional Euclidean distance between the locations.
 60  */
 61 jsts.geom.Coordinate.prototype.distance = function(p) {
 62   var dx = this.x - p.x;
 63   var dy = this.y - p.y;
 64 
 65   return Math.sqrt(dx * dx + dy * dy);
 66 };

 69 /**
 70  * Returns whether the planar projections of the two <code>Coordinate</code>s
 71  * are equal.
 72  *
 73  * @param {Coordinate}
 74  *          other a <code>Coordinate</code> with which to do the 2D
 75  *          comparison.
 76  * @return {boolean} <code>true</code> if the x- and y-coordinates are equal;
 77  *         the z-coordinates do not have to be equal.
 78  */
 79 jsts.geom.Coordinate.prototype.equals2D = function(other) {
 80   if (this.x !== other.x) {
 81     return false;
 82   }
 83 
 84   if (this.y !== other.y) {
 85     return false;
 86   }
 87 
 88   return true;
 89 };

 92 /**
 93  * Returns <code>true</code> if <code>other</code> has the same values for
 94  * the x and y ordinates. Since Coordinates are 2.5D, this routine ignores the z
 95  * value when making the comparison.
 96  *
 97  * @param {Coordinate}
 98  *          other a <code>Coordinate</code> with which to do the comparison.
 99  * @return {boolean} <code>true</code> if <code>other</code> is a
100  *         <code>Coordinate</code> with the same values for the x and y
101  *         ordinates.
102  */
103 jsts.geom.Coordinate.prototype.equals = function(other) {
104   if (!other instanceof jsts.geom.Coordinate || other === undefined) {
105     return false;
106   }
107   return this.equals2D(other);
108 };

111 /**
112  * Compares this {@link Coordinate} with the specified {@link Coordinate} for
113  * order. This method ignores the z value when making the comparison. Returns:
114  * <UL>
115  * <LI> -1 : this.x < other.x || ((this.x == other.x) && (this.y < other.y))
116  * <LI> 0 : this.x == other.x && this.y = other.y
117  * <LI> 1 : this.x > other.x || ((this.x == other.x) && (this.y > other.y))
118  *
119  * </UL>
120  * Note: This method assumes that ordinate values are valid numbers. NaN values
121  * are not handled correctly.
122  *
123  * @param {Coordinate}
124  *          other the <code>Coordinate</code> with which this
125  *          <code>Coordinate</code> is being compared.
126  * @return {Boolean} -1, zero, or 1 as explained above.
127  */
128 jsts.geom.Coordinate.prototype.compareTo = function(other) {
129   if (this.x < other.x) {
130     return -1;
131   }
132   if (this.x > other.x) {
133     return 1;
134   }
135   if (this.y < other.y) {
136     return -1;
137   }
138   if (this.y > other.y) {
139     return 1;
140   }
141 
142   return 0;
143 };

227 /**
228  * @param {Point}
229  *          other point to compare.
230  * @param {double}
231  *          tolerance tolerance used in comparison.
232  * @return {Boolean} true if gemetries match.
233  */
234 jsts.geom.Coordinate.prototype.equalsExact = function(other, tolerance) {
235   if (this.CLASS_NAME !== other.CLASS_NAME) {
236     return false;
237   }
238   if (this.isEmpty() && other.isEmpty()) {
239     return true;
240   }
241   return jsts.geom.Geometry.prototype.equal(other, this, tolerance);
242 };

265 OpenLayers.Geometry.Point = OpenLayers.Class(OpenLayers.Geometry.Point,
266     jsts.geom.Coordinate, {
267       // replace constructor with one that also accepts JTS arguments
268       initialize: function(x, y) {
269         OpenLayers.Geometry.prototype.initialize.apply(this, arguments);
270         if (x === undefined || x === null) {
271           this.x = 0;
272           this.y = 0;
273         } else if (typeof x === 'number' || typeof x === 'string') {
274           this.x = parseFloat(x);
275           this.y = parseFloat(y);
276         } else if (x instanceof jsts.geom.Coordinate) {
277           y = x.y;
278           x = x.x;
279           this.x = parseFloat(x);
280           this.y = parseFloat(y);
281         }
282       }
283     });
