  7 /**
  8  * A lightweight class used to store coordinates on the 2-dimensional Cartesian
  9  * plane. It is distinct from {@link Point}, which is a subclass of
 10  * {@link Geometry}. Unlike objects of type {@link Point} (which contain
 11  * additional information such as an envelope, a precision model, and spatial
 12  * reference system information), a <code>Coordinate</code> only contains
 13  * coordinate values and accessor methods.
 14  *
 15  * @requires jsts/geom/Geometry.js
 16  */
 17 
 18 
 19 
 20 /**
 21  * @constructor
 22  * @augments OpenLayers.Geometry.Point
 23  * @augments jsts.geom.Geometry
 24  */
 25 jsts.geom.Coordinate = function() {
 26 };
 27 
 28 
 29 /**
 30  * Sets this <code>Coordinate</code>s (x,y,z) values to that of
 31  * <code>other</code>.
 32  *
 33  * @param {Coordinate}
 34  *          other the <code>Coordinate</code> to copy.
 35  */
 36 jsts.geom.Coordinate.prototype.setCoordinate = function(other) {
 37   this.x = other.x;
 38   this.y = other.y;
 39 };
 40 
 41 
 42 /**
 43  * Clones this instance.
 44  *
 45  * @return {jsts.geom.Coordinate} A point instance cloned from this.
 46  */
 47 jsts.geom.Coordinate.prototype.clone = function() {
 48   return new jsts.geom.Coordinate(this.x, this.y);
 49 };
 50 
 51 
 52 /**
 53  * Computes the 2-dimensional Euclidean distance to another location. The
 54  * Z-ordinate is ignored.
 55  *
 56  * @param {Coordinate}
 57  *          p a point.
 58  * @return {double} the 2-dimensional Euclidean distance between the locations.
 59  */
 60 jsts.geom.Coordinate.prototype.distance = function(p) {
 61   var dx = this.x - p.x;
 62   var dy = this.y - p.y;
 63 
 64   return Math.sqrt(dx * dx + dy * dy);
 65 };

 68 /**
 69  * Returns whether the planar projections of the two <code>Coordinate</code>s
 70  * are equal.
 71  *
 72  * @param {Coordinate}
 73  *          other a <code>Coordinate</code> with which to do the 2D
 74  *          comparison.
 75  * @return {boolean} <code>true</code> if the x- and y-coordinates are equal;
 76  *         the z-coordinates do not have to be equal.
 77  */
 78 jsts.geom.Coordinate.prototype.equals2D = function(other) {
 79   if (this.x !== other.x) {
 80     return false;
 81   }
 82 
 83   if (this.y !== other.y) {
 84     return false;
 85   }
 86 
 87   return true;
 88 };
 89 
 90 
 91 /**
 92  * Returns <code>true</code> if <code>other</code> has the same values for
 93  * the x and y ordinates. Since Coordinates are 2.5D, this routine ignores the z
 94  * value when making the comparison.
 95  *
 96  * @param {Coordinate}
 97  *          other a <code>Coordinate</code> with which to do the comparison.
 98  * @return {boolean} <code>true</code> if <code>other</code> is a
 99  *         <code>Coordinate</code> with the same values for the x and y
100  *         ordinates.
101  */
102 jsts.geom.Coordinate.prototype.equals = function(other) {
103   if (!other instanceof jsts.geom.Coordinate || other === undefined) {
104     return false;
105   }
106   return this.equals2D(other);
107 };
108

110 /**
111  * Compares this {@link Coordinate} with the specified {@link Coordinate} for
112  * order. This method ignores the z value when making the comparison. Returns:
113  * <UL>
114  * <LI> -1 : this.x < other.x || ((this.x == other.x) && (this.y < other.y))
115  * <LI> 0 : this.x == other.x && this.y = other.y
116  * <LI> 1 : this.x > other.x || ((this.x == other.x) && (this.y > other.y))
117  *
118  * </UL>
119  * Note: This method assumes that ordinate values are valid numbers. NaN values
120  * are not handled correctly.
121  *
122  * @param {Coordinate}
123  *          other the <code>Coordinate</code> with which this
124  *          <code>Coordinate</code> is being compared.
125  * @return {Boolean} -1, zero, or 1 as explained above.
126  */
127 jsts.geom.Coordinate.prototype.compareTo = function(other) {
128   if (this.x < other.x) {
129     return -1;
130   }
131   if (this.x > other.x) {
132     return 1;
133   }
134   if (this.y < other.y) {
135     return -1;
136   }
137   if (this.y > other.y) {
138     return 1;
139   }
140 
141   return 0;
142 };
143 
144 jsts.geom.Coordinate.prototype.hashCode = function() {
145   return '' + this.x + this.y;
146 };

149 OpenLayers.Geometry.Point = OpenLayers.Class(OpenLayers.Geometry.Point,
150     jsts.geom.Coordinate, {
151       // replace constructor with one that also accepts JTS arguments
152       initialize: function(x, y) {
153         OpenLayers.Geometry.prototype.initialize.apply(this, arguments);
154         if (x === undefined || x === null) {
155           this.x = 0;
156           this.y = 0;
157         } else if (typeof x === 'number' || typeof x === 'string') {
158           this.x = parseFloat(x);
159           this.y = parseFloat(y);
160         } else if (x instanceof jsts.geom.Coordinate) {
161           y = x.y;
162           x = x.x;
163           this.x = parseFloat(x);
164           this.y = parseFloat(y);
165         }
166       }
167     });
168 jsts.geom.Coordinate = OpenLayers.Geometry.Point;
