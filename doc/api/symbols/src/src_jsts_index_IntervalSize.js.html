  3 /**
  4  * Provides a test for whether an interval is so small it should be considered
  5  * as zero for the purposes of inserting it into a binary tree. The reason this
  6  * check is necessary is that round-off error can cause the algorithm used to
  7  * subdivide an interval to fail, by computing a midpoint value which does not
  8  * lie strictly between the endpoints.
  9  *
 10  */
 11 jsts.index.IntervalSize = OpenLayers.Class();
 12 
 13 
 14 /**
 15  * This value is chosen to be a few powers of 2 less than the number of bits
 16  * available in the double representation (i.e. 53). This should allow enough
 17  * extra precision for simple computations to be correct, at least for
 18  * comparison purposes.
 19  */
 20 jsts.index.IntervalSize.prototype.MIN_BINARY_EXPONENT = -50;
 21 
 22 
 23 /**
 24  * Computes whether the interval [min, max] is effectively zero width. I.e. the
 25  * width of the interval is so much less than the location of the interval that
 26  * the midpoint of the interval cannot be represented precisely.
 27  *
 28  * @param {Number}
 29  *          min the min-value in the interval.
 30  * @param {Number}
 31  *          max the max-value in the interval.
 32  * @return {Boolean} true if the interval should be considered zero.
 33  */
 34 jsts.index.IntervalSize.prototype.isZeroWidth = function(min, max) {
 35   var width = max - min;
 36   if (width === 0.0) {
 37     return true;
 38   }
 39 
 40   var maxAbs, scaledInterval, level;
 41   maxAbs = Math.max(Math.abs(min), Math.abs(max));
 42   scaledInterval = width / maxAbs;
 43 
 44   level = jsts.index.DoubleBits.prototype.exponent(scaledInterval);
 45   return level <= jsts.index.IntervalSize.prototype.MIN_BINARY_EXPONENT;
 46 };
