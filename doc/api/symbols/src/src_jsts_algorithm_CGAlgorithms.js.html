 57 /**
 58  * Returns the index of the direction of the point <code>q</code>
 59  * relative to a
 60  * vector specified by <code>p1-p2</code>.
 61  *
 62  * @param {jsts.geom.Coordinate}
 63  *        p1 the origin point of the vector.
 64  * @param {jsts.geom.Coordinate}
 65  *        p2 the final point of the vector.
 66  * @param {jsts.geom.Coordinate}
 67  *        q the point to compute the direction to.
 68  *
 69  * @return {Number}
 70  *         1 if q is counter-clockwise (left) from p1-p2.
 71  * @return {Number}
 72  *         -1 if q is clockwise (right) from p1-p2.
 73  * @return {Number}
 74  *         0 if q is collinear with p1-p2.
 75  */
 76 jsts.algorithm.CGAlgorithms.orientationIndex = function(p1, p2, q) {
 77   /**
 78    * MD - 9 Aug 2010
 79    * It seems that the basic algorithm is slightly orientation dependent,
 80    * when computing the orientation of a point very close to a line.
 81    * This is possibly due to the arithmetic in the translation to the origin.
 82    *
 83    * For instance, the following situation produces identical results
 84    * in spite of the inverse orientation of the line segment:
 85    *
 86    * Coordinate p0 = new Coordinate(219.3649559090992, 140.84159161824724);
 87    * Coordinate p1 = new Coordinate(168.9018919682399, -5.713787599646864);
 88    *
 89    * Coordinate p = new Coordinate(186.80814046338352, 46.28973405831556);
 90    * int orient = orientationIndex(p0, p1, p);
 91    * int orientInv = orientationIndex(p1, p0, p);
 92 
 93    * A way to force consistent results is to normalize the orientation of
 94    * the vector using the following code.
 95    * However, this may make the results of orientationIndex inconsistent
 96    * through the triangle of points, so it's not clear this is
 97    * an appropriate patch.
 98    *
 99    */
100 
101   var dx1, dy1, dx2, dy2;
102   dx1 = p2.x - p1.x;
103   dy1 = p2.y - p1.y;
104   dx2 = q.x - p2.x;
105   dy2 = q.y - p2.y;
106 
107   return jsts.algorithm.RobustDeterminant.signOfDet2x2(dx1, dy1, dx2, dy2);
108 };

157 /**
158  * Tests whether a point lies on the line segments defined by a
159  * list of coordinates.
160  *
161  * @param  {jsts.geom.Coordinate}
162  *         p the coordinate to test.
163  * @param  {Array{jsts.geom.Coordinate}}
164  *         pt An array of coordinates defining line segments
165  * @return {Boolean}
166  *         true if the point is a vertex of the line
167  *         or lies in the interior of a line segment in the linestring.
168  */
169 jsts.algorithm.CGAlgorithms.isOnLine = function(p, pt) {
170   var lineIntersector, i, il, p0, p1;
171   lineIntersector = new jsts.algorithm.RobustLineIntersector();
172 
173   i = 1, il = pt.length;
174   for (i; i < il; i++) {
175     p0 = pt[i - 1];
176     p1 = pt[i];
177     lineIntersector.computeIntersection(p, p0, p1);
178 
179     if (lineIntersector.hasIntersection()) {
180       return true;
181     }
182   }
183   return false;
184 };

187 /**
188  * Computes whether a ring defined by an array of {@link Coordinate}s is
189  * oriented counter-clockwise.
190  * <ul>
191  * <li>The list of points is assumed to have the first and last points equal.
192  * <li>This will handle coordinate lists which contain repeated points.
193  * </ul>
194  * This algorithm is <b>only</b> guaranteed to work with valid rings.
195  * If the ring is invalid (e.g. self-crosses or touches),
196  * the computed result may not be correct.
197  *
198  * @param {Array{jsts.geom.Coordinate}}
199  *        ring an array of Coordinates forming a ring
200  * @return {Boolean}
201  *         true if the ring is oriented counter-clockwise.
202  * @throws IllegalArgumentException if there are too few points to determine orientation (< 3)
203  */
204 jsts.algorithm.CGAlgorithms.isCCW = function(ring) {
205   var nPts, hiPt, hiIndex, p, iPrev, iNext, prev, next, i, disc, isCCW;
206 
207   //# of points without closing endpoint
208   nPts = ring.length - 1;
209 
210   //sanity check
211   if (nPts < 3) {
212     throw new jsts.IllegalArgumentError('Ring has fewer than 3 points, so orientation cannot be determined');
213   }
214 
215   //find highets point
216   hiPt = ring[0];
217   hiIndex = 0;
218 
219   i = 1;
220   for (i; i <= nPts; i++) {
221     p = ring[i];
222     if (p.y > hiPt.y) {
223       hiPt = p;
224       hiIndex = i;
225     }
226   }
227 
228   //find distinct point before highest point
229   iPrev = hiIndex;
230   do {
231     iPrev = iPrev - 1;
232     if (iPrev < 0) {
233       iPrev = nPts;
234     }
235   }while (ring[iPrev].equals2D(hiPt) && iPrev !== hiIndex);
236 
237   //find distinct point after highest point
238   iNext = hiIndex;
239   do {
240     iNext = (iNext + 1) % nPts;
241   }while (ring[iNext].equals2D(hiPt) && iNext !== hiIndex);
242 
243   prev = ring[iPrev];
244   next = ring[iNext];
245 
246   /**
247    * This check catches cases where the ring contains an A-B-A configuration of points.
248    * This can happen if the ring does not contain 3 distinct points
249    * (including the case where the input array has fewer than 4 elements),
250    * or it contains coincident line segments.
251    */
252   if (prev.equals2D(hiPt) || next.equals2D(hiPt) || prev.equals2D(next)) {
253     return false;
254   }
255 
256   disc = jsts.algorithm.CGAlgorithms.computeOrientation(prev, hiPt, next);
257 
258   /**
259    *  If disc is exactly 0, lines are collinear.  There are two possible cases:
260    *  (1) the lines lie along the x axis in opposite directions
261    *  (2) the lines lie on top of one another
262    *
263    *  (1) is handled by checking if next is left of prev ==> CCW
264    *  (2) will never happen if the ring is valid, so don't check for it
265    *  (Might want to assert this)
266    */
267   isCCW = false;
268   if (disc === 0) {
269     // poly is CCW if prev x is right of next x
270     isCCW = (prev.x > next.x);
271   }
272   else {
273     // if area is positive, points are ordered CCW
274     isCCW = (disc > 0);
275   }
276 
277   return isCCW;
278 };

305 /**
306  * Computes the distance from a point p to a line segment AB
307  *
308  * Note: NON-ROBUST!
309  *
310  * @param {jsts.geom.Coordinate}
311  *        p the point to compute the distance for.
312  * @param {jsts.geom.Coordinate}
313  *        A one point of the line.
314  * @param {jsts.geom.Coordinate}
315  *        B another point of the line (must be different to A).
316  * @return {Number}
317  *         the distance from p to line segment AB.
318  */
319 jsts.algorithm.CGAlgorithms.distancePointLine = function(p, A, B) {
320   //if start = end, then just compute distance to one of the endpoints
321   if (A.x === B.x && A.y === B.y) {
322     return p.distance(A);
323   }
324   //otherwise use comp.graphics.algorithms Frequently Asked Questions method
325   /*(1)             AC dot AB
326                    r = ---------
327                          ||AB||^2
328     r has the following meaning:
329     r=0 P = A
330     r=1 P = B
331     r<0 P is on the backward extension of AB
332     r>1 P is on the forward extension of AB
333     0<r<1 P is interior to AB
334   */
335   var r, s;
336   r = ((p.x - A.x) * (B.x - A.x) + (p.y - A.y) * (B.y - A.y)) /
337       ((B.x - A.x) * (B.x - A.x) + (B.y - A.y) * (B.y - A.y));
338 
339   if (r <= 0.0) {
340     return p.distance(A);
341   }
342   if (r >= 1.0) {
343     return p.distance(B);
344   }
345 
346   /*(2)
347     (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)
348   s = -----------------------------
349              L^2
350 
351   Then the distance from C to P = |s|*L.
352   */
353 
354   s = ((A.y - p.y) * (B.x - A.x) - (A.x - p.x) * (B.y - A.y)) /
355       ((B.x - A.x) * (B.x - A.x) + (B.y - A.y) * (B.y - A.y));
356 
357   return Math.abs(s) *
358       Math.sqrt(((B.x - A.x) * (B.x - A.x) + (B.y - A.y) * (B.y - A.y)));
359 };

362 /**
363  * Computes the perpendicular distance from a point p
364  * to the (infinite) line containing the points AB
365  *
366  * @param {jsts.geom.Coordinate}
367  *        p the point to compute the distance for.
368  * @param {jsts.geom.Coordinate}
369  *        A one point of the line.
370  * @param {jsts.geom.Coordinate}
371  *        B another point of the line (must be different to A).
372  * @return {Number}
373  *         the distance from p to line AB.
374  */
375 jsts.algorithm.CGAlgorithms.distancePointLinePerpendicular = function(p, A, B) {
376   //use comp.graphics.algorithms Frequently Asked Questions method
377   /*(2)
378                    (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)
379               s = -----------------------------
380                                    L^2
381 
382               Then the distance from C to P = |s|*L.
383   */
384   var s = ((A.y - p.y) * (B.x - A.x) - (A.x - p.x) * (B.y - A.y)) /
385       ((B.x - A.x) * (B.x - A.x) + (B.y - A.y) * (B.y - A.y));
386 
387   return Math.abs(s) *
388       Math.sqrt(((B.x - A.x) * (B.x - A.x) + (B.y - A.y) * (B.y - A.y)));
389 };

392 /**
393  * Computes the distance from a point to a sequence
394  * of line segments.
395  *
396  * @param {jsts.geom.Coordinate}
397  *        p a point.
398  * @param {Array{jsts.geom.Coordinate}}
399  *        line a sequence of contiguous line segments defined by their vertices
400  * @return {Number}
401  *         the minimum distance between the point and the line segments.
402  */
403 jsts.algorithm.CGAlgorithms.distancePointLine = function(p, line) {
404   if (line.length === 0) {
405     throw new jsts.IllegalArgumentError('Line array must contain at least one vertex');
406   }
407   var minDistance, i, il, dist;
408   i = 0, il = line.length - 1;
409 
410   minDistance = p.distance(line[0]);
411   for (i; i < il; i++) {
412     dist = jsts.algorithm.CGAlgorithms.distancePointLine(p, line[i], line[i + 1]);
413     if (dist < minDistance) {
414       minDistance = dist;
415     }
416   }
417   return minDistance;
418 };

420 /**
421  * Computes the distance from a line segment AB to a line segment CD
422  *
423  * Note: NON-ROBUST!
424  *
425  * @param {jsts.geom.Coordinate}
426  *        A a point of one line.
427  * @param {jsts.geom.Coordinate}
428  *        B the second point of  (must be different to A).
429  * @param {jsts.geom.Coordinate}
430  *        C one point of the line.
431  * @param {jsts.geom.Coordinate}
432  *        D another point of the line (must be different to A).
433  * @return {Number}
434  *         the distance.
435  */
436 
437 jsts.algorithm.CGAlgorithms.distanceLineLine = function(A, B, C, D) {
438   //check for zero-length segments
439   if (A.equals(B)) {
440     return jsts.algorithm.CGAlgorithms.distancePointLine(A, C, D);
441   }
442   if (C.equals(D)) {
443     return jsts.algorithm.CGAlgorithms.distancePointLine(D, A, B);
444   }
445 
446   //AB and CD are line segments
447   /* from comp.graphics.algo
448 
449   Solving the above for r and s yields
450         (Ay-Cy)(Dx-Cx)-(Ax-Cx)(Dy-Cy)
451              r = ----------------------------- (eqn 1)
452         (Bx-Ax)(Dy-Cy)-(By-Ay)(Dx-Cx)
453 
454       (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)
455     s = ----------------------------- (eqn 2)
456       (Bx-Ax)(Dy-Cy)-(By-Ay)(Dx-Cx)
457   Let P be the position vector of the intersection point, then
458     P=A+r(B-A) or
459     Px=Ax+r(Bx-Ax)
460     Py=Ay+r(By-Ay)
461   By examining the values of r & s, you can also determine some other
462   limiting conditions:
463     If 0<=r<=1 & 0<=s<=1, intersection exists
464     r<0 or r>1 or s<0 or s>1 line segments do not intersect
465     If the denominator in eqn 1 is zero, AB & CD are parallel
466     If the numerator in eqn 1 is also zero, AB & CD are collinear.
467 
468   */
469   var r_top, r_bot, s_top, s_bot, s, r;
470   r_top = (A.y - C.y) * (D.x - C.x) - (A.x - C.x) * (D.y - C.y);
471   r_bot = (B.x - A.x) * (D.y - C.y) - (B.y - A.y) * (D.x - C.x);
472 
473   s_top = (A.y - C.y) * (B.x - A.x) - (A.x - C.x) * (B.y - A.y);
474   s_bot = (B.x - A.x) * (D.y - C.y) - (B.y - A.y) * (D.x - C.x);
475 
476 
477   if ((r_bot == 0) || (s_bot == 0)) {
478     return Math.min(distancePointLine(A, C, D),
479         Math.min(distancePointLine(B, C, D),
480             Math.min(distancePointLine(C, A, B),
481                 distancePointLine(D, A, B))));
482   }
483 
484   s = s_top / s_bot;
485   r = r_top / r_bot;
486   if ((r < 0) || (r > 1) || (s < 0) || (s > 1)) {
487     //no intersection
488     return;
489     Math.min(distancePointLine(A, C, D),
490         Math.min(distancePointLine(B, C, D),
491         Math.min(distancePointLine(C, A, B),
492         distancePointLine(D, A, B))));
493   }
494 
495   return 0.0; //intersection exists
496 };

499 /**
500  * Computes the signed area for a ring.
501  * The signed area is positive if
502  * the ring is oriented CW, negative if the ring is oriented CCW,
503  * and zero if the ring is degenerate or flat.
504  *
505  * @param {Array{jsts.geom.Coordinate}}
506  *        ring the coordinates forming the ring
507  * @return {Number}
508  *         the signed area of the ring.
509  */
510 jsts.algorithm.CGAlgorithms.signedArea = function(ring) {
511   if (ring.length < 3) {
512     return 0.0;
513   }
514   var sum, i, il, bx, by, cx, cy;
515 
516   sum = 0.0, i = 0, il = ring.length - 1;
517 
518   for (i; i < il; i++) {
519     bx = ring[i].x;
520     by = ring[i].y;
521     cx = ring[i + 1].x;
522     cy = ring[i + 1].y;
523     sum += (bx + cx) * (cy - by);
524   }
525 
526   return -sum / 2.0;
527 };

530 /**
531  * Computes the signed area for a ring.
532  * The signed area is:
533  * <ul>
534  * <li>positive if the ring is oriented CW
535  * <li>negative if the ring is oriented CCW
536  * <li>zero if the ring is degenerate or flat
537  * </ul>
538  *
539  * @param {Array{jsts.geom.Coordinate}}
540  *        ring the coordinates forming the ring
541  * @return {Number}
542  *         the signed area of the ring.
543  */
544 jsts.algorithm.CGAlgorithms.signedArea = function(ring) {
545   var n, sum, p, bx, by, i, cx, cy;
546 
547   n = ring.length;
548   if (n < 3) {
549     return 0.0;
550   }
551 
552   sum = 0.0;
553   p = ring[0];
554 
555   bx = p.x;
556   by = p.y;
557 
558   i = 1;
559   for (i; i < n; i++) {
560     p = ring[i];
561     cx = p.x;
562     cy = p.y;
563     sum += (bx + cx) * (cy - by);
564     bx = cx;
565     by = cy;
566   }
567 
568   return -sum / 2.0;
569 };
