  9 /**
 10  * A ConnectedElementPointFilter extracts a single point from each connected
 11  * element in a Geometry (e.g. a polygon, linestring or point) and returns them
 12  * in a list. The elements of the list are
 13  * {@link com.vividsolutions.jts.operation.distance.GeometryLocation}s.
 14  *
 15  * @param {[]}
 16  *          locations
 17  * @augments jsts.geom.GeometryFilter
 18  * @constructor
 19  */
 20 jsts.operation.distance.ConnectedElementLocationFilter = function(locations) {
 21   this.locations = locations;
 22 };
 23 
 24 jsts.operation.distance.ConnectedElementLocationFilter.prototype = new jsts.geom.GeometryFilter();
 25 
 26 
 27 /**
 28  * @type {[]}
 29  * @private
 30  */
 31 jsts.operation.distance.ConnectedElementLocationFilter.prototype.locations = null;
 32 
 33 
 34 /**
 35  * Returns a list containing a point from each Polygon, LineString, and Point
 36  * found inside the specified geometry. Thus, if the specified geometry is not a
 37  * GeometryCollection, an empty list will be returned. The elements of the list
 38  * are {@link com.vividsolutions.jts.operation.distance.GeometryLocation}s.
 39  *
 40  * @param {Geometry}
 41  *          geom
 42  * @return {[]}
 43  */
 44 jsts.operation.distance.ConnectedElementLocationFilter.getLocations = function(
 45     geom) {
 46   var locations = [];
 47   geom.apply(new jsts.operation.distance.ConnectedElementLocationFilter(
 48       locations));
 49   return locations;
 50 };
 51 
 52 
 53 /**
 54  * @param {Geometry}
 55  *          geom
 56  */
 57 jsts.operation.distance.ConnectedElementLocationFilter.prototype.filter = function(
 58     geom) {
 59   if (geom instanceof jsts.geom.Point || geom instanceof jsts.geom.LineString ||
 60       geom instanceof jsts.geom.Polygon)
 61     this.locations.push(new jsts.operation.distance.GeometryLocation(geom, 0,
 62         geom.getCoordinate()));
 63 };
