  9 /**
 10  * Provides constants representing the dimensions of a point, a curve and a
 11  * surface. Also provides constants representing the dimensions of the empty
 12  * geometry and non-empty geometries, and the wildcard constant
 13  * {@link #DONTCARE} meaning "any dimension". These constants are used as the
 14  * entries in {@link IntersectionMatrix}s.
 15  *
 16  * @constructor
 17  */
 18 jsts.geom.Dimension = function() {
 19 };
 20 
 21 
 22 /**
 23  * Dimension value of a point (0).
 24  *
 25  * @type {int}
 26  */
 27 jsts.geom.Dimension.P = 0;
 28 
 29 
 30 /**
 31  * Dimension value of a curve (1).
 32  *
 33  * @type {int}
 34  */
 35 jsts.geom.Dimension.L = 1;
 36 
 37 
 38 /**
 39  * Dimension value of a surface (2).
 40  *
 41  * @type {int}
 42  */
 43 jsts.geom.Dimension.A = 2;
 44 
 45 
 46 /**
 47  * Dimension value of the empty geometry (-1).
 48  *
 49  * @type {int}
 50  */
 51 jsts.geom.Dimension.FALSE = -1;
 52 
 53 
 54 /**
 55  * Dimension value of non-empty geometries (= {P, L, A}).
 56  *
 57  * @type {int}
 58  */
 59 jsts.geom.Dimension.TRUE = -2;
 60 
 61 
 62 /**
 63  * Dimension value for any dimension (= {FALSE, TRUE}).
 64  *
 65  * @type {int}
 66  */
 67 jsts.geom.Dimension.DONTCARE = -3;
 68 
 69 
 70 /**
 71  * Converts the dimension value to a dimension symbol, for example,
 72  * <code>TRUE => 'T'</code> .
 73  *
 74  * @param {int}
 75  *          dimensionValue a number that can be stored in the
 76  *          <code>IntersectionMatrix</code> . Possible values are
 77  *          <code>{TRUE, FALSE, DONTCARE, 0, 1, 2}</code>.
 78  * @return {String} a character for use in the string representation of an
 79  *         <code>IntersectionMatrix</code>. Possible values are
 80  *         <code>{T, F, * , 0, 1, 2}</code> .
 81  */
 82 jsts.geom.Dimension.toDimensionSymbol = function(dimensionValue) {
 83   switch (dimensionValue) {
 84     case jsts.geom.Dimension.FALSE:
 85       return 'F';
 86     case jsts.geom.Dimension.TRUE:
 87       return 'T';
 88     case jsts.geom.Dimension.DONTCARE:
 89       return '*';
 90     case jsts.geom.Dimension.P:
 91       return '0';
 92     case jsts.geom.Dimension.L:
 93       return '1';
 94     case jsts.geom.Dimension.A:
 95       return '2';
 96   }
 97   throw new jsts.IllegalArgumentError('Unknown dimension value: ' +
 98       dimensionValue);
 99 };
100 
101 
102 /**
103  * Converts the dimension symbol to a dimension value, for example,
104  * <code>'*' => DONTCARE</code> .
105  *
106  * @param {string}
107  *          dimensionSymbol a character for use in the string representation of
108  *          an <code>IntersectionMatrix</code>. Possible values are
109  *          <code>{T, F, * , 0, 1, 2}</code> .
110  * @return {int} a number that can be stored in the
111  *         <code>IntersectionMatrix</code> . Possible values are
112  *         <code>{TRUE, FALSE, DONTCARE, 0, 1, 2}</code>.
113  */
114 jsts.geom.Dimension.toDimensionValue = function(dimensionSymbol) {
115   switch (dimensionSymbol.toUpperCase()) {
116     case 'F':
117       return jsts.geom.Dimension.FALSE;
118     case 'T':
119       return jsts.geom.Dimension.TRUE;
120     case '*':
121       return jsts.geom.Dimension.DONTCARE;
122     case '0':
123       return jsts.geom.Dimension.P;
124     case '1':
125       return jsts.geom.Dimension.L;
126     case '2':
127       return jsts.geom.Dimension.A;
128   }
129   throw new jsts.error.IllegalArgumentError('Unknown dimension symbol: ' +
130       dimensionSymbol);
131 };
