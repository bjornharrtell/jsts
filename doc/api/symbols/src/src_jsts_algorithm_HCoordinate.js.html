  9 /**
 10  * Represents a homogeneous coordinate in a 2-D coordinate space. In JTS
 11  * {@link HCoordinate}s are used as a clean way of computing intersections
 12  * between line segments.
 13  *
 14  * Initializes a new HCoordinate using the OpenLayers inheritance mechanism.
 15  * Will call correct init* function depending on argument.
 16  *
 17  * @constructor
 18  */
 19 jsts.algorithm.HCoordinate = function() {
 20   this.x = 0.0;
 21   this.y = 0.0;
 22   this.w = 1.0;
 23 
 24   if (arguments.length === 1) {
 25     this.initFrom1Coordinate(arguments[0]);
 26   } else if (arguments.length === 2 &&
 27       arguments[0] instanceof jsts.geom.Coordinate) {
 28     this.initFrom2Coordinates(arguments[0], arguments[1]);
 29   } else if (arguments.length === 2 &&
 30       arguments[0] instanceof jsts.algorithm.HCoordinate) {
 31     this.initFrom2HCoordinates(arguments[0], arguments[1]);
 32   } else if (arguments.length === 2) {
 33     this.initFromXY(arguments[0], arguments[1]);
 34   } else if (arguments.length === 3) {
 35     this.initFromXYW(arguments[0], arguments[1], arguments[2]);
 36   } else if (arguments.length === 4) {
 37     this.initFromXYW(arguments[0], arguments[1], arguments[2], arguments[3]);
 38   }
 39 };

 42 /**
 43  * Computes the (approximate) intersection point between two line segments using
 44  * homogeneous coordinates.
 45  * <p>
 46  * Note that this algorithm is not numerically stable; i.e. it can produce
 47  * intersection points which lie outside the envelope of the line segments
 48  * themselves. In order to increase the precision of the calculation input
 49  * points should be normalized before passing them to this routine.
 50  *
 51  * @param {jsts.geom.Coordinate}
 52  *          p1 first coordinate for the first line.
 53  * @param {jsts.geom.Coordinate}
 54  *          p2 second coordinate for the first line.
 55  * @param {jsts.geom.Coordinate}
 56  *          q1 first coordinate for the second line.
 57  * @param {jsts.geom.Coordinate}
 58  *          q2 second coordinate for the second line.
 59  * @return {jsts.geom.Coordinate} The coordinate of the intersection.
 60  */
 61 jsts.algorithm.HCoordinate.intersection = function(p1, p2, q1, q2) {
 62   var px, py, pw, qx, qy, qw, x, y, w, xInt, yInt;
 63 
 64   // unrolled computation
 65   px = p1.y - p2.y;
 66   py = p2.x - p1.x;
 67   pw = p1.x * p2.y - p2.x * p1.y;
 68 
 69   qx = q1.y - q2.y;
 70   qy = q2.x - q1.x;
 71   qw = q1.x * q2.y - q2.x * q1.y;
 72 
 73   x = py * qw - qy * pw;
 74   y = qx * pw - px * qw;
 75   w = px * qy - qx * py;
 76 
 77   xInt = x / w;
 78   yInt = y / w;
 79 
 80   if (!isFinite(xInt) || !isFinite(yInt)) {
 81     throw new jsts.error.NotRepresentableException();
 82   }
 83 
 84   return new jsts.geom.Coordinate(xInt, yInt);
 85 };

 88 /**
 89  * Initializes a new HCoordinate from 1 Coordinate
 90  *
 91  * @param {jsts.geom.Coordinate}
 92  *          p the coordinate.
 93  */
 94 jsts.algorithm.HCoordinate.prototype.initFrom1Coordinate = function(p) {
 95   this.x = p.x;
 96   this.y = p.y;
 97   this.w = 1.0;
 98 };

101 /**
102  * Constructs a homogeneous coordinate which is the intersection of the lines
103  * define by the homogenous coordinates represented by two {@link Coordinate}s.
104  *
105  * @param {jsts.geom.Coordinate}
106  *          p1 the first coordinate.
107  * @param {jsts.geom.Coordinate}
108  *          p2 the second coordinate.
109  */
110 jsts.algorithm.HCoordinate.prototype.initFrom2Coordinates = function(p1, p2) {
111   // optimization when it is known that w = 1
112   this.x = p1.y - p2.y;
113   this.y = p2.x - p1.x;
114   this.w = p1.x * p2.y - p2.x * p1.y;
115 };
116 
117 
118 /**
119  * Initializes from 2 HCoordinates
120  *
121  * @param {jsts.algorithm.HCoordinate}
122  *          p1 the first HCoordinate.
123  * @param {jsts.algorithm.HCoordinate}
124  *          p2 the second HCoordinate.
125  */
126 jsts.algorithm.HCoordinate.prototype.initFrom2HCoordinates = function(p1, p2) {
127   this.x = p1.y * p2.w - p2.y * p1.w;
128   this.y = p2.x * p1.w - p1.x * p2.w;
129   this.w = p1.x * p2.y - p2.x * p1.y;
130 };
131 
132 
133 /**
134  * Initializes from x,y,w
135  *
136  * @param {Number}
137  *          x the x-value.
138  * @param {Number}
139  *          y the y-value.
140  * @param {Number}
141  *          w the w-value.
142  */
143 jsts.algorithm.HCoordinate.prototype.initFromXYW = function(x, y, w) {
144   this.x = x;
145   this.y = y;
146   this.w = w;
147 };
148 
149 
150 /**
151  * Initializes from x,y
152  *
153  * @param {Number}
154  *          x the x-value.
155  * @param {Number}
156  *          y the y-value.
157  */
158 jsts.algorithm.HCoordinate.prototype.initFromXY = function(x, y) {
159   this.x = x;
160   this.y = y;
161   this.w = 1.0;
162 };

165 /**
166  * Initializes from 4 Coordinates
167  *
168  * @param {jsts.geom.Coordinate}
169  *          p1 the first coordinate.
170  * @param {jsts.geom.Coordinate}
171  *          p2 the second coordinate.
172  * @param {jsts.geom.Coordinate}
173  *          q1 the first coordinate.
174  * @param {jsts.geom.Coordinate}
175  *          q2 the second coordinate.
176  */
177 jsts.algorithm.HCoordinate.prototype.initFrom4Coordinates = function(p1, p2,
178     q1, q2) {
179   var px, py, pw, qx, qy, qw;
180   // unrolled computation
181   px = p1.y - p2.y;
182   py = p2.x - p1.x;
183   pw = p1.x * p2.y - p2.x * p1.y;
184 
185   qx = q1.y - q2.y;
186   qy = q2.x - q1.x;
187   qw = q1.x * q2.y - q2.x * q1.y;
188 
189   this.x = py * qw - qy * pw;
190   this.y = qx * pw - px * qw;
191   this.w = px * qy - qx * py;
192 };

195 /**
196  * Gets x/w
197  *
198  * @return {Number} x/w.
199  */
200 jsts.algorithm.HCoordinate.prototype.getX = function() {
201   var a = this.x / this.w;
202 
203   if (!isFinite(a)) {
204     throw new jsts.error.NotRepresentableException();
205   }
206   return a;
207 };
208 
209 
210 /**
211  * Gets y/w
212  *
213  * @return {Number} y/w.
214  */
215 jsts.algorithm.HCoordinate.prototype.getY = function() {
216   var a = this.y / this.w;
217 
218   if (!isFinite(a)) {
219     throw new jsts.error.NotRepresentableException();
220   }
221   return a;
222 };
223 
224 
225 /**
226  * Gets a coordinate represented by this HCoordinate
227  *
228  * @return {jst.geom.Coordinate} The coordinate.
229  */
230 jsts.algorithm.HCoordinate.prototype.getCoordinate = function() {
231   var p = new jsts.geom.Coordinate();
232   p.x = this.getX();
233   p.y = this.getY();
234   return p;
235 };
