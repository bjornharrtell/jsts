  6 /**
  7  * Unions a collection of Geometry or a single Geometry
  8  * (which may be a collection) together.
  9  * By using this special-purpose operation over a collection of geometries
 10  * it is possible to take advantage of various optimizations to improve performance.
 11  * Heterogeneous {@link GeometryCollection}s are fully supported.
 12  * <p>
 13  * The result obeys the following contract:
 14  * <ul>
 15  * <li>Unioning a set of overlapping {@link Polygons}s has the effect of
 16  * merging the areas (i.e. the same effect as
 17  * iteratively unioning all individual polygons together).
 18  *
 19  * <li>Unioning a set of {@link LineString}s has the effect of <b>fully noding</b>
 20  * and <b>dissolving</b> the input linework.
 21  * In this context "fully noded" means that there will be a node or endpoint in the output
 22  * for every endpoint or line segment crossing in the input.
 23  * "Dissolved" means that any duplicate (e.g. coincident) line segments or portions
 24  * of line segments will be reduced to a single line segment in the output.
 25  * This is consistent with the semantics of the
 26  * {@link Geometry#union(Geometry)} operation.
 27  * If <b>merged</b> linework is required, the {@link LineMerger} class can be used.
 28  *
 29  * <li>Unioning a set of {@link Points}s has the effect of merging
 30  * al identical points (producing a set with no duplicates).
 31  * </ul>
 32  *
 33  * <tt>UnaryUnion</tt> always operates on the individual components of MultiGeometries.
 34  * So it is possible to use it to "clean" invalid self-intersecting MultiPolygons
 35  * (although the polygon components must all still be individually valid.)
 36  *
 37  */
 38

 53 /**
 54  *
 55  * @param {Geometry|Geometry[]} geoms a Geometry or Geometry collection.
 56  * @param {Geometryfactory} [gemFact] a GeometryFactory.
 57  * @return {Geometry}
 58  */
 59 jsts.operation.union.UnaryUnionOp.union = function(geoms, gemFact) {
 60   var op = new jsts.operation.union.UnaryUnionOp(geoms, gemFact);
 61   return op.union();
 62 };

 89 /**
 90  * @param {Geometry|Geometry[]} geoms a Geometry or Geometry collection.
 91  */
 92 jsts.operation.union.UnaryUnionOp.prototype.extract = function(geoms) {
 93   if (geoms instanceof Array) {
 94     for (var i = 0, l = geoms.length; i < l; i++) {
 95       this.extract(geoms[i]);
 96     }
 97   }
 98   else {
 99     if (this.geomFact === null) {
100       geomFact = geoms.getFactory();
101     }
102     jsts.geom.util.GeometryExtractor.extract(geoms, jsts.geom.Polygon, this.polygons);
103     jsts.geom.util.GeometryExtractor.extract(geoms, jsts.geom.LineString, this.lines);
104     jsts.geom.util.GeometryExtractor.extract(geoms, jsts.geom.Point, this.points);
105   }
106 };
107 
108 
109 /**
110  * Gets the union of the input geometries.
111  * If no input geometries were provided, a POINT EMPTY is returned.
112  *
113  * @return {Geometry} a Geometry containing the union.
114  * @return {GeomatryCollection} an empty GEOMETRYCOLLECTION if no geometries were provided in the input.
115  */
116 jsts.operation.union.UnaryUnionOp.prototype.union = function() {
117   if (this.geomFact === null) {
118     return null;
119   }
120 
121   /**
122    * For points and lines, only a single union operation is
123    * required, since the OGC model allowings self-intersecting
124    * MultiPoint and MultiLineStrings.
125    * This is not the case for polygons, so Cascaded Union is required.
126    */
127 
128   var unionPoints = null;
129   if (this.points.length > 0) {
130     var ptGeom = this.geomFact.buildGeometry(this.points);
131     unionPoints = this.unionNoOpt(ptGeom);
132   }
133 
134   var unionLines = null;
135   if (this.lines.length > 0) {
136     var lineGeom = this.geomFact.buildGeometry(this.lines);
137     unionLines = this.unionNoOpt(lineGeom);
138   }
139 
140   var unionPolygons = null;
141   if (this.polygons.length > 0) {
142     unionPolygons = jsts.operation.union.CascadedPolygonUnion.union(this.polygons);
143   }
144 
145   /**
146    * Performing two unions is somewhat inefficient,
147    * but is mitigated by unioning lines and points first
148    */
149 
150   var unionLA = this.unionWithNull(unionLines, unionPolygons);
151   var union = null;
152   if (unionPoints === null) {
153     union = unionLA;
154   }
155   else if (unionLA === null) {
156     union = unionPoints;
157   }
158   else {
159     union = jsts.operation.union.PointGeometryUnion(unionPoints, unionLA);
160   }
161 
162   if (union === null) {
163     return this.geomFact.createGeometryCollection(null);
164   }
165 
166   return union;
167 };

170 /**
171  * Computes the union of two geometries,
172  * either of both of which may be null.
173  *
174  * @param {Geometry} g0 a Geometry.
175  * @param {Geometry} g1 a Geometry.
176  * @return {Geometry} the union of the input(s).
177  * @return {null} null if both inputs are null.
178  * @private
179  */
180 jsts.operation.union.UnaryUnionOp.prototype.unionWithNull = function(g0, g1) {
181   if (g0 === null && g1 === null) {
182     return null;
183   }
184   if (g1 == null) {
185     return g0;
186   }
187   if (g0 === null) {
188     return g1;
189   }
190   return g0.union(g1);
191 };
