  9 /**
 10  * Extracts all the 1-dimensional ({@link LineString}) components from a
 11  * {@link Geometry}.
 12  *
 13  * @augments GeometryComponentFilter
 14  * @constructor
 15  */
 16 jsts.geom.util.LinearComponentExtracter = function(lines, isForcedToLineString) {
 17   this.lines = lines;
 18   this.isForcedToLineString = isForcedToLineString;
 19 };

 21 jsts.geom.util.LinearComponentExtracter.prototype = new jsts.geom.GeometryComponentFilter();

 36 /**
 37  * Extracts the linear components from a single {@link Geometry} and adds them
 38  * to the provided {@link Collection}.
 39  *
 40  * NOTE: will call "overloaded" function depending on args.
 41  *
 42  * @param {[]}
 43  *          geoms the collection of geometries from which to extract linear
 44  *          components.
 45  * @param {[]}
 46  *          lines the collection to add the extracted linear components to.
 47  * @return {[]} the collection of linear components (LineStrings or
 48  *         LinearRings).
 49  */
 50 jsts.geom.util.LinearComponentExtracter.getLines = function(geoms, lines) {
 51   if (arguments.length == 1) {
 52     return jsts.geom.util.LinearComponentExtracter.getLines5.apply(this, arguments);
 53   }
 54   else if (arguments.length == 2 && typeof lines === 'boolean') {
 55     return jsts.geom.util.LinearComponentExtracter.getLines6.apply(this, arguments);
 56   }
 57   else if (arguments.length == 2 && geoms instanceof jsts.geom.Geometry) {
 58     return jsts.geom.util.LinearComponentExtracter.getLines3.apply(this, arguments);
 59   }
 60   else if (arguments.length == 3 && geoms instanceof jsts.geom.Geometry) {
 61     return jsts.geom.util.LinearComponentExtracter.getLines4.apply(this, arguments);
 62   }
 63   else if (arguments.length == 3) {
 64     return jsts.geom.util.LinearComponentExtracter.getLines2.apply(this, arguments);
 65   }
 66 
 67   for (var i = 0; i < geoms.length; i++) {
 68     var g = geoms[i];
 69     jsts.geom.util.LinearComponentExtracter.getLines3(g, lines);
 70   }
 71   return lines;
 72 };

 75 /**
 76  * Extracts the linear components from a single {@link Geometry} and adds them
 77  * to the provided {@link Collection}.
 78  *
 79  * @param {[]}
 80  *          geoms the Collection of geometries from which to extract linear
 81  *          components.
 82  * @param {[]}
 83  *          lines the collection to add the extracted linear components to.
 84  * @param {boolean}
 85  *          forceToLineString true if LinearRings should be converted to
 86  *          LineStrings.
 87  * @return {[]} the collection of linear components (LineStrings or
 88  *         LinearRings).
 89  */
 90 jsts.geom.util.LinearComponentExtracter.getLines2 = function(geoms, lines,
 91     forceToLineString) {
 92   for (var i = 0; i < geoms.length; i++) {
 93     var g = geoms[i];
 94     jsts.geom.util.LinearComponentExtracter.getLines4(g, lines,
 95         forceToLineString);
 96   }
 97   return lines;
 98 };

101 /**
102  * Extracts the linear components from a single {@link Geometry} and adds them
103  * to the provided {@link Collection}.
104  *
105  * @param {Geometry}
106  *          geom the geometry from which to extract linear components.
107  * @param {[]}
108  *          lines the Collection to add the extracted linear components to.
109  * @return {[]} the Collection of linear components (LineStrings or
110  *         LinearRings).
111  */
112 jsts.geom.util.LinearComponentExtracter.getLines3 = function(geom, lines) {
113   if (geom instanceof LineString) {
114     lines.add(geom);
115   } else {
116     geom.apply(new jsts.geom.util.LinearComponentExtracter(lines));
117   }
118   return lines;
119 };

150  *          geom the geometry from which to extract linear components.

153 jsts.geom.util.LinearComponentExtracter.getLines5 = function(geom) {

158 /**
159  * Extracts the linear components from a single geometry. If more than one
160  * geometry is to be processed, it is more efficient to create a single
161  * {@link LinearComponentExtracter} instance and pass it to multiple geometries.
162  *
163  * @param {Geometry}
164  *          geom the geometry from which to extract linear components.
165  * @param {boolean}
166  *          forceToLineString true if LinearRings should be converted to
167  *          LineStrings.
168  * @return {[] the list of linear components.}
169  */
170 jsts.geom.util.LinearComponentExtracter.getLines6 = function(geom,
171     forceToLineString) {
172   var lines = [];
173   geom.apply(new jsts.geom.util.LinearComponentExtracter(lines,
174       forceToLineString));
175   return lines;
176 };

179 /**
180  * Indicates that LinearRing components should be converted to pure LineStrings.
181  *
182  * @param {boolean}
183  *          isForcedToLineString true if LinearRings should be converted to
184  *          LineStrings.
185  */
186 jsts.geom.util.LinearComponentExtracter.prototype.setForceToLineString = function(
187     isForcedToLineString) {
188   this.isForcedToLineString = isForcedToLineString;
189 };
190 
191 jsts.geom.util.LinearComponentExtracter.prototype.filter = function(geom) {
192   if (this.isForcedToLineString && geom instanceof jsts.geom.LinearRing) {
193     var line = geom.getFactory().createLineString(geom.getCoordinateSequence());
194     this.lines.push(line);
195     return;
196   }
197   // if not being forced, and this is a linear component
198   // NOTE: inheritance will not show LinearRing to be of LineString heritance...
199   if (geom instanceof jsts.geom.LineString || geom instanceof jsts.geom.LinearRing)
200     this.lines.push(geom);
201 
202   // else this is not a linear component, so skip it
203 };
