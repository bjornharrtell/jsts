 82 /**
 83  * Returns the angle
 84  * Calls correct angle* depending on argument
 85  *
 86  * @return {Number}
 87  *          The angle in radians.
 88  */
 89 jsts.algorithm.Angle.angle = function() {
 90   if (arguments.length === 1) {
 91     return jsts.algorithm.Angle.angleFromOrigo(arguments[0]);
 92   }else {
 93     return jsts.algorithm.Angle.angleBetweenCoords(arguments[0], arguments[1]);
 94   }
 95 };

 98 /**
 99  * Returns the angle of the vector from p0 to p1,
100  * relative to the positive X-axis.
101  * The angle is normalized to be in the range [ -Pi, Pi ].
102  *
103  * @param {jsts.geom.Coordinate}
104  *        p0 a coordinate.
105  * @param {jsts.geom.Coordinate}
106  *        p1 a coordinate.
107  * @return {Number}
108  *         the normalized angle (in radians) that p0-p1 makes with the positive
109  *         x-axis.
110  */
111 jsts.algorithm.Angle.angleBetweenCoords = function(p0, p1) {
112   var dx, dy;
113   dx = p1.x - p0.x;
114   dy = p1.y - p0.y;
115   return Math.atan2(dy, dx);
116 };

119 /**
120  * Returns the angle that the vector from (0,0) to p,
121  * relative to the positive X-axis.
122  * The angle is normalized to be in the range ( -Pi, Pi ].
123  *
124  * @param {jsts.geom.Coordinate}
125  *        p a coordinate.
126  * @return {Number}
127  *         the normalized angle (in radians) that p makes with the positive
128  *         x-axis.
129  */
130 jsts.algorithm.Angle.angleFromOrigo = function(p) {
131   return Math.atan2(p.y, p.x);

135 /**
136  * Tests whether the angle between p0-p1-p2 is acute.
137  * An angle is acute if it is less than 90 degrees.
138  * <p>
139  * Note: this implementation is not precise (determistic) for angles very close to 90 degrees.
140  *
141  * @param {jsts.geom.Coordinate}
142  *        p0 an endpoint of the angle.
143  * @param {jsts.geom.Coordinate}
144  *        p1 the base of the angle.
145  * @param {jsts.geom.Coordinate}
146  *        p2 the other endpoint of the angle.
147  * @return {Boolean}
148  *         true if the angle is acute.
149  */
150 jsts.algorithm.Angle.isAcute = function(p0, p1, p2) {
151   var dx0, dy0, dx1, dy1, dotprod;
152 
153   //relies on fact that A dot B is positive if A ang B is acute
154   dx0 = p0.x - p1.x;
155   dy0 = p0.y - p1.y;
156   dx1 = p2.x - p1.x;
157   dy1 = p2.y - p1.y;
158   dotprod = dx0 * dx1 + dy0 * dy1;
159   return dotprod > 0;
160 };

163 /**
164  * Tests whether the angle between p0-p1-p2 is obtuse.
165  * An angle is obtuse if it is greater than 90 degrees.
166  * <p>
167  * Note: this implementation is not precise (determistic) for angles very close to 90 degrees.
168  *
169  * @param {jsts.geom.Coordinate}
170  *        p0 an endpoint of the angle.
171  * @param {jsts.geom.Coordinate}
172  *        p1 the base of the angle.
173  * @param {jsts.geom.Coordinate}
174  *        p2 the other endpoint of the angle.
175  * @return {Boolean}
176  *         true if the angle is obtuse.
177  */
178 jsts.algorithm.Angle.isObtuse = function(p0, p1, p2) {
179   var dx0, dy0, dx1, dy1, dotprod;
180 
181   //relies on fact that A dot B is negative iff A ang B is obtuse
182   dx0 = p0.x - p1.x;
183   dy0 = p0.y - p1.y;
184   dx1 = p2.x - p1.x;
185   dy1 = p2.y - p1.y;
186   dotprod = dx0 * dx1 + dy0 * dy1;
187   return dotprod < 0;
188 };

191 /**
192  * Returns the unoriented smallest angle between two vectors.
193  * The computed angle will be in the range [0, Pi).
194  *
195  * @param {jsts.geom.Coordinate}
196  *        tip1 the tip of one vector.
197  * @param {jsts.geom.Coordinate}
198  *        tail the tail of each vector.
199  * @param {jsts.geom.Coordinate}
200  *        tip2 the tip of the other vector.
201  * @return {Number}
202  *         the angle between tail-tip1 and tail-tip2.
203  */
204 jsts.algorithm.Angle.angleBetween = function(tip1, tail, tip2) {
205   var a1, a2;
206   a1 = jsts.algorithm.Angle.angle(tail, tip1);
207   a2 = jsts.algorithm.Angle.angle(tail, tip2);
208 
209   return jsts.algorithm.Angle.diff(a1, a2);
210 };

213 /**
214  * Returns the oriented smallest angle between two vectors.
215  * The computed angle will be in the range (-Pi, Pi].
216  * A positive result corresponds to a counterclockwise rotation
217  * from v1 to v2;
218  * a negative result corresponds to a clockwise rotation.
219  *
220  * @param {jsts.geom.Coordinate}
221  *        tip1 the tip of v1.
222  * @param {jsts.geom.Coordinate}
223  *        tail the tail of each vector.
224  * @param {jsts.geom.Coordinate}
225  *        tip2 the tip of v2.
226  * @return {Number}
227  *         the angle between v1 and v2, relative to v1.
228  */
229 jsts.algorithm.Angle.angleBetweenOriented = function(tip1, tail, tip2) {
230   var a1, a2, angDel;
231 
232   a1 = jsts.algorithm.Angle.angle(tail, tip1);
233   a2 = jsts.algorithm.Angle.angle(tail, tip2);
234   angDel = a2 - a1;
235 
236   // normalize, maintaining orientation
237   if (angDel <= -Math.PI) {
238     return angDel + jsts.algorithm.Angle.PI_TIMES_2;
239   }
240   if (angDel > Math.PI) {
241     return angDel - jsts.algorithm.Angle.PI_TIMES_2;
242   }
243   return angDel;
244 };

247 /**
248  * Computes the interior angle between two segments of a ring. The ring is
249  * assumed to be oriented in a clockwise direction. The computed angle will be
250  * in the range [0, 2Pi]
251  *
252  * @param {jsts.geom.Coordinate}
253  *        p0 a point of the ring.
254  * @param {jsts.geom.Coordinate}
255  *        p1 the next point of the ring.
256  * @param {jsts.geom.Coordinate}
257  *        p2 the next point of the ring.
258  * @return {Number}
259  *         the interior angle based at <code>p1.</code>
260  */
261 jsts.algorithm.Angle.interiorAngle = function(p0, p1, p2) {
262   var anglePrev, angleNext;
263 
264   anglePrev = jsts.algorithm.Angle.angle(p1, p0);
265   angleNext = jsts.algorithm.Angle.angle(p1, p2);
266   return Math.abs(angleNext - anglePrev);
267 };

270 /**
271  * Returns whether an angle must turn clockwise or counterclockwise
272  * to overlap another angle.
273  *
274  * @param {Number}
275  *        ang1 an angle (in radians).
276  * @param {Number}
277  *        ang2 an angle (in radians).
278  * @return {Number}
279  *         whether a1 must turn CLOCKWISE, COUNTERCLOCKWISE or NONE to
280  *         overlap a2.
281  */
282 jsts.algorithm.Angle.getTurn = function(ang1, ang2) {
283   var crossproduct = Math.sin(ang2 - ang1);
284 
285   if (crossproduct > 0) {
286     return jsts.algorithm.Angle.COUNTERCLOCKWISE;
287   }
288   if (crossproduct < 0) {
289     return jsts.algorithm.Angle.CLOCKWISE;
290   }
291   return jsts.algorithm.Angle.NONE;
292 };

295 /**
296  * Computes the normalized value of an angle, which is the
297  * equivalent angle in the range ( -Pi, Pi ].
298  *
299  * @param {Number}
300  *        angle the angle to normalize.
301  * @return {Number}
302  *         an equivalent angle in the range (-Pi, Pi].
303  */
304 jsts.algorithm.Angle.normalize = function(angle) {
305   while (angle > Math.PI) {
306     angle -= jsts.algorithm.Angle.PI_TIMES_2;
307   }
308   while (angle <= -Math.PI) {
309     angle += jsts.algorithm.Angle.PI_TIMES_2;
310   }
311   return angle;
312 };

315 /**
316  * Computes the normalized positive value of an angle, which is the
317  * equivalent angle in the range [ 0, 2*Pi ).
318  * E.g.:
319  * <ul>
320  * <li>normalizePositive(0.0) = 0.0
321  * <li>normalizePositive(-PI) = PI
322  * <li>normalizePositive(-2PI) = 0.0
323  * <li>normalizePositive(-3PI) = PI
324  * <li>normalizePositive(-4PI) = 0
325  * <li>normalizePositive(PI) = PI
326  * <li>normalizePositive(2PI) = 0.0
327  * <li>normalizePositive(3PI) = PI
328  * <li>normalizePositive(4PI) = 0.0
329  * </ul>
330  *
331  * @param {Number}
332  *        angle the angle to normalize, in radians.
333  * @return {Number}
334  *         an equivalent positive angle.
335  */
336 jsts.algorithm.Angle.normalizePositive = function(angle) {
337   if (angle < 0.0) {
338     while (angle < 0.0) {
339       angle += jsts.algorithm.Angle.PI_TIMES_2;
340     }
341     // in case round-off error bumps the value over
342     if (angle >= jsts.algorithm.Angle.PI_TIMES_2) {
343       angle = 0.0;
344     }
345   }
346   else {
347     while (angle >= jsts.algorithm.Angle.PI_TIMES_2) {
348       angle -= jsts.algorithm.Angle.PI_TIMES_2;
349     }
350     // in case round-off error bumps the value under
351     if (angle < 0.0) {
352       angle = 0.0;
353     }
354   }
355   return angle;
356 };

359 /**
360  * Computes the unoriented smallest difference between two angles.
361  * The angles are assumed to be normalized to the range [-Pi, Pi].
362  * The result will be in the range [0, Pi].
363  *
364  * @param {Number}
365  *        ang1 the angle of one vector (in [-Pi, Pi] ).
366  * @param {Number}
367  *        ang2 the angle of the other vector (in range [-Pi, Pi] ).
368  * @return {Number}
369  *         the angle (in radians) between the two vectors (in range [0, Pi] ).
370  */
371 jsts.algorithm.Angle.diff = function(ang1, ang2) {
372   var delAngle;
373 
374   if (ang1 < ang2) {
375     delAngle = ang2 - ang1;
376   } else {
377     delAngle = ang1 - ang2;
378   }
379 
380   if (delAngle > Math.PI) {
381     delAngle = (2 * Math.PI) - delAngle;
382   }
383 
384   return delAngle;
385 };
