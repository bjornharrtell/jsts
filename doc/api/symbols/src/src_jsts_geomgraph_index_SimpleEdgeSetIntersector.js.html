  9 /**
 10  * Finds all intersections in one or two sets of edges, using the
 11  * straightforward method of comparing all segments. This algorithm is too slow
 12  * for production use, but is useful for testing purposes.
 13  *
 14  * @constructor
 15  */
 16 jsts.geomgraph.index.SimpleEdgeSetIntersector = function() {
 17 
 18 };
 19 
 20 jsts.geomgraph.index.SimpleEdgeSetIntersector.prototype = new jsts.geomgraph.index.EdgeSetIntersector();
 21 
 22 
 23 /**
 24  * statistics information
 25  *
 26  * @type {int}
 27  */
 28 jsts.geomgraph.index.SimpleEdgeSetIntersector.prototype.nOverlaps = 0;
 29 
 30 
 31 /**
 32  * @param {[]}
 33  *          edges
 34  * @param {SegmentIntersector}
 35  *          si
 36  * @param {boolean}
 37  *          testAllSegments
 38  */
 39 jsts.geomgraph.index.SimpleEdgeSetIntersector.prototype.computeIntersections = function(
 40     edges, si, testAllSegments) {
 41   this.nOverlaps = 0;
 42 
 43   for (var i0 = 0; i0 < edges.length; i0++) {
 44     var edge0 = edges[i0];
 45     for (var i1 = 0; i1 < edges.length; i1++) {
 46       var edge1 = edges[i1];
 47       if (testAllSegments || edge0 != edge1) {
 48         this.computeIntersects(edge0, edge1, si);
 49       }
 50     }
 51   }
 52 };
 53 
 54 
 55 /**
 56  * @param {[]}
 57  *          edges0
 58  * @param {[]}
 59  *          edges1
 60  * @param {SegmentIntersector}
 61  *          si
 62  */
 63 jsts.geomgraph.index.SimpleEdgeSetIntersector.prototype.computeIntersections = function(
 64     edges0, edges1, si) {
 65   this.nOverlaps = 0;
 66 
 67   for (var i0 = 0; i0 < edges0.length; i0++) {
 68     var edge0 = edges0[i0];
 69     for (var i1 = 0; i1 < edges1.length; i1++) {
 70       var edge1 = edges1[i0];
 71       this.computeIntersects(edge0, edge1, si);
 72     }
 73   }
 74 };
 75 
 76 
 77 /**
 78  * Performs a brute-force comparison of every segment in each Edge. This has n^2
 79  * performance, and is about 100 times slower than using monotone chains.
 80  *
 81  * @param {Edge}
 82  *          e0
 83  * @param {Edge}
 84  *          e1
 85  * @param {SegmentIntersector}
 86  *          si
 87  */
 88 jsts.geomgraph.index.SimpleEdgeSetIntersector.prototype.computeIntersects = function(
 89     e0, e1, si) {
 90   var pts0 = e0.getCoordinates();
 91   var pts1 = e1.getCoordinates();
 92   for (var i0 = 0; i0 < pts0.length - 1; i0++) {
 93     for (var i1 = 0; i1 < pts1.length - 1; i1++) {
 94       si.addIntersections(e0, i0, e1, i1);
 95     }
 96   }
 97 };
