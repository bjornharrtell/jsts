 13 /**
 14  * Finds all intersections in one or two sets of edges, using the
 15  * straightforward method of comparing all segments. This algorithm is too slow
 16  * for production use, but is useful for testing purposes.
 17  *
 18  * @constructor
 19  */
 20 jsts.geomgraph.index.SimpleEdgeSetIntersector = function() {
 21 
 22 };
 23 
 24 jsts.geomgraph.index.SimpleEdgeSetIntersector.prototype = new jsts.geomgraph.index.EdgeSetIntersector();
 25 
 26 
 27 /**
 28  * statistics information
 29  *
 30  * @type {int}
 31  */
 32 jsts.geomgraph.index.SimpleEdgeSetIntersector.prototype.nOverlaps = 0;
 33 
 34 
 35 /**
 36  * @param {[]}
 37  *          edges
 38  * @param {SegmentIntersector}
 39  *          si
 40  * @param {boolean}
 41  *          testAllSegments
 42  */
 43 jsts.geomgraph.index.SimpleEdgeSetIntersector.prototype.computeIntersections = function(
 44     edges, si, testAllSegments) {
 45 
 46   if (si instanceof Array) {
 47     this.computeIntersections2.apply(this, arguments);
 48     return;
 49   }
 50 
 51   var i0, i1, edge0, edge1;
 52 
 53   this.nOverlaps = 0;
 54 
 55   for (i0 = 0; i0 < edges.length; i0++) {
 56     edge0 = edges[i0];
 57     for (i1 = 0; i1 < edges.length; i1++) {
 58       edge1 = edges[i1];
 59       if (testAllSegments || edge0 !== edge1) {
 60         this.computeIntersects(edge0, edge1, si);
 61       }
 62     }
 63   }
 64 };
 65 
 66 
 67 /**
 68  * @param {[]}
 69  *          edges0
 70  * @param {[]}
 71  *          edges1
 72  * @param {SegmentIntersector}
 73  *          si
 74  */
 75 jsts.geomgraph.index.SimpleEdgeSetIntersector.prototype.computeIntersections2 = function(
 76     edges0, edges1, si) {
 77   this.nOverlaps = 0;
 78 
 79   for (var i0 = 0; i0 < edges0.length; i0++) {
 80     var edge0 = edges0[i0];
 81     for (var i1 = 0; i1 < edges1.length; i1++) {
 82       var edge1 = edges1[i1];
 83       this.computeIntersects(edge0, edge1, si);
 84     }
 85   }
 86 };

 89 /**
 90  * Performs a brute-force comparison of every segment in each Edge. This has n^2
 91  * performance, and is about 100 times slower than using monotone chains.
 92  *
 93  * @param {Edge}
 94  *          e0
 95  * @param {Edge}
 96  *          e1
 97  * @param {SegmentIntersector}
 98  *          si
 99  */
100 jsts.geomgraph.index.SimpleEdgeSetIntersector.prototype.computeIntersects = function(
101     e0, e1, si) {
102   var pts0 = e0.getCoordinates();
103   var pts1 = e1.getCoordinates();
104   var i0, i1;
105   for (i0 = 0; i0 < pts0.length - 1; i0++) {
106     for (i1 = 0; i1 < pts1.length - 1; i1++) {
107       si.addIntersections(e0, i0, e1, i1);
108     }
109   }
110 };
