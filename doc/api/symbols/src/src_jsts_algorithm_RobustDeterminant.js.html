 36 /**
 37  * Computes the sign of the determinant of the 2x2 matrix
 38  * with the given entries, in a robust way.
 39  *
 40  * @param {Number}
 41  *        x1 X-1.
 42  * @param {Number}
 43  *        y1 Y-1.
 44  * @param {Number}
 45  *        x2 X-2.
 46  * @param {Number}
 47  *        y2 Y-1.
 48  *
 49  * @return {Number}
 50  *         -1 if the determinant is negative,.
 51  * @return {Number}
 52  *         1 if the determinant is positive,.
 53  * @return {Number}
 54  *         0 if the determinant is 0.
 55  */
 56 jsts.algorithm.RobustDeterminant.signOfDet2x2 = function(x1, y1, x2, y2) {
 57   //returns -1 if the determinant is negative,
 58   // returns  1 if the determinant is positive,
 59   // retunrs  0 if the determinant is null.
 60   var sign, swap, k, count;
 61   count = 0;
 62 
 63   sign = 1;
 64 
 65   /*
 66    *  testing null entries
 67    */
 68   if ((x1 == 0.0) || (y2 == 0.0)) {
 69     if ((y1 == 0.0) || (x2 == 0.0)) {
 70       return 0;
 71     }
 72     else if (y1 > 0) {
 73       if (x2 > 0) {
 74         return -sign;
 75       }
 76       else {
 77         return sign;
 78       }
 79     }
 80     else {
 81       if (x2 > 0) {
 82         return sign;
 83       }
 84       else {
 85         return -sign;
 86       }
 87     }
 88   }
 89   if ((y1 == 0.0) || (x2 == 0.0)) {
 90     if (y2 > 0) {
 91       if (x1 > 0) {
 92         return sign;
 93       }
 94       else {
 95         return -sign;
 96       }
 97     }
 98     else {
 99       if (x1 > 0) {
100         return -sign;
101       }
102       else {
103         return sign;
104       }
105     }
106   }
107 
108   /*
109    *  making y coordinates positive and permuting the entries
110    */
111   /*
112    *  so that y2 is the biggest one
113    */
114   if (0.0 < y1) {
115     if (0.0 < y2) {
116       if (y1 <= y2) {
117       }
118       else {
119         sign = -sign;
120         swap = x1;
121         x1 = x2;
122         x2 = swap;
123         swap = y1;
124         y1 = y2;
125         y2 = swap;
126       }
127     }
128     else {
129       if (y1 <= -y2) {
130         sign = -sign;
131         x2 = -x2;
132         y2 = -y2;
133       }
134       else {
135         swap = x1;
136         x1 = -x2;
137         x2 = swap;
138         swap = y1;
139         y1 = -y2;
140         y2 = swap;
141       }
142     }
143   }
144   else {
145     if (0.0 < y2) {
146       if (-y1 <= y2) {
147         sign = -sign;
148         x1 = -x1;
149         y1 = -y1;
150       }
151       else {
152         swap = -x1;
153         x1 = x2;
154         x2 = swap;
155         swap = -y1;
156         y1 = y2;
157         y2 = swap;
158       }
159     }
160     else {
161       if (y1 >= y2) {
162         x1 = -x1;
163         y1 = -y1;
164         x2 = -x2;
165         y2 = -y2;
166       }
167       else {
168         sign = -sign;
169         swap = -x1;
170         x1 = -x2;
171         x2 = swap;
172         swap = -y1;
173         y1 = -y2;
174         y2 = swap;
175       }
176     }
177   }
178 
179   /*
180    *  making x coordinates positive
181    */
182   /*
183    *  if |x2| < |x1| one can conclude
184    */
185   if (0.0 < x1) {
186     if (0.0 < x2) {
187       if (x1 <= x2) {
188       }
189       else {
190         return sign;
191       }
192     }
193     else {
194       return sign;
195     }
196   }
197   else {
198     if (0.0 < x2) {
199       return -sign;
200     }
201     else {
202       if (x1 >= x2) {
203         sign = -sign;
204         x1 = -x1;
205         x2 = -x2;
206       }
207       else {
208         return -sign;
209       }
210     }
211   }
212 
213   /*
214    *  all entries strictly positive   x1 <= x2 and y1 <= y2
215    */
216   while (true) {
217     count = count + 1;
218     // MD - UNSAFE HACK for testing only!
219     //    k = (int) (x2 / x1);
220     k = Math.floor(x2 / x1);
221     x2 = x2 - k * x1;
222     y2 = y2 - k * y1;
223 
224     /*
225      *  testing if R (new U2) is in U1 rectangle
226      */
227     if (y2 < 0.0) {
228       return -sign;
229     }
230     if (y2 > y1) {
231       return sign;
232     }
233 
234     /*
235      *  finding R'
236      */
237     if (x1 > x2 + x2) {
238       if (y1 < y2 + y2) {
239         return sign;
240       }
241     }
242     else {
243       if (y1 > y2 + y2) {
244         return -sign;
245       }
246       else {
247         x2 = x1 - x2;
248         y2 = y1 - y2;
249         sign = -sign;
250       }
251     }
252     if (y2 == 0.0) {
253       if (x2 == 0.0) {
254         return 0;
255       }
256       else {
257         return -sign;
258       }
259     }
260     if (x2 == 0.0) {
261       return sign;
262     }
263 
264     /*
265      *  exchange 1 and 2 role.
266      */
267     // MD - UNSAFE HACK for testing only!
268     //    k = (int) (x1 / x2);
269     k = Math.floor(x1 / x2);
270     x1 = x1 - k * x2;
271     y1 = y1 - k * y2;
272 
273     /*
274      *  testing if R (new U1) is in U2 rectangle
275      */
276     if (y1 < 0.0) {
277       return sign;
278     }
279     if (y1 > y2) {
280       return -sign;
281     }
282 
283     /*
284      *  finding R'
285      */
286     if (x2 > x1 + x1) {
287       if (y2 < y1 + y1) {
288         return -sign;
289       }
290     }
291     else {
292       if (y2 > y1 + y1) {
293         return sign;
294       }
295       else {
296         x1 = x2 - x1;
297         y1 = y2 - y1;
298         sign = -sign;
299       }
300     }
301     if (y1 == 0.0) {
302       if (x1 == 0.0) {
303         return 0;
304       }
305       else {
306         return sign;
307       }
308     }
309     if (x1 == 0.0) {
310       return -sign;
311     }
312   }
313 };
