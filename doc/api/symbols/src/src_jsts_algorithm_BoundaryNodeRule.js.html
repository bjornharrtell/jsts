 45 /**
 46  * A {@link BoundaryNodeRule} specifies that points are in the boundary of a
 47  * lineal geometry iff the point lies on the boundary of an odd number of

 49  * {@link LineString}s have an empty boundary.
 50  * <p>
 51  * This is the rule specified by the <i>OGC SFS</i>, and is the default rule

 58 jsts.algorithm.Mod2BoundaryNodeRule.prototype = new jsts.algorithm.BoundaryNodeRule();
 59 
 60 jsts.algorithm.Mod2BoundaryNodeRule.prototype.isInBoundary = function(
 61     boundaryCount) {
 62   // the "Mod-2 Rule"
 63   return boundaryCount % 2 == 1;
 64 };
