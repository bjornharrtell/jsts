  9 /**
 10  * The base class for operations that require {@link GeometryGraph}s.
 11  *
 12  * @param {Geometry}
 13  *          g0
 14  * @param {Geometry}
 15  *          g1
 16  * @param {BoundaryNodeRule}
 17  *          boundaryNodeRule
 18  * @constructor
 19  */
 20 jsts.operation.GeometryGraphOperation = function(g0, g1, boundaryNodeRule) {
 21   this.li = new jsts.algorithm.RobustLineIntersector();
 22 
 23   if (g0 === undefined) {
 24     return;
 25   }
 26 
 27   if (g1 === undefined) {
 28     this.setComputationPrecision(g0.getPrecisionModel());
 29 
 30     arg[0] = new jsts.geomgraph.GeometryGraph(0, g0);
 31     return;
 32   }
 33 
 34   boundaryNodeRule = boundaryNodeRule ||
 35       jsts.algorithm.BoundaryNodeRule.OGC_SFS_BOUNDARY_RULE;
 36 
 37   // use the most precise model for the result
 38   if (g0.getPrecisionModel().compareTo(g1.getPrecisionModel()) >= 0)
 39     this.setComputationPrecision(g0.getPrecisionModel());
 40   else
 41     this.setComputationPrecision(g1.getPrecisionModel());
 42 
 43   this.arg[0] = new jsts.geomgraph.GeometryGraph(0, g0, boundaryNodeRule);
 44   this.arg[1] = new jsts.geomgraph.GeometryGraph(1, g1, boundaryNodeRule);
 45 };
 46 
 47 
 48 /**
 49  * @type {LineIntersector}
 50  * @protected
 51  */
 52 jsts.operation.GeometryGraphOperation.prototype.li = null;
 53 
 54 
 55 /**
 56  * @type {PrecisionModel}
 57  * @protected
 58  */
 59 jsts.operation.GeometryGraphOperation.prototype.resultPrecisionModel = null;
 60 
 61 
 62 /**
 63  * The operation args into an array so they can be accessed by index
 64  *
 65  * @type {GeometryGraph[]}
 66  * @protected
 67  */
 68 jsts.operation.GeometryGraphOperation.prototype.arg = [];
 69 
 70 
 71 /**
 72  * @param {int}
 73  *          i
 74  * @return {Geometry}
 75  */
 76 jsts.operation.GeometryGraphOperation.prototype.getArgGeometry = function(i) {
 77   return arg[i].getGeometry();
 78 };
 79 
 80 
 81 /**
 82  * @param {PrecisionModel}
 83  *          pm
 84  * @protected
 85  */
 86 jsts.operation.GeometryGraphOperation.prototype.setComputationPrecision = function(pm) {
 87   this.resultPrecisionModel = pm;
 88   this.li.setPrecisionModel(this.resultPrecisionModel);
 89 };
