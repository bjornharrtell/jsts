 12 /**
 13  * A GeometryGraph is a graph that models a given Geometry
 14  *
 15  * @param {int}
 16  *          argIndex
 17  * @param {Geometry}
 18  *          parentGeom
 19  * @param {BoundaryNodeRule}
 20  *          boundaryNodeRule
 21  * @augments jsts.planargraph.PlanarGraph
 22  */
 23 jsts.geomgraph.GeometryGraph = function(argIndex, parentGeom, boundaryNodeRule) {
 24   jsts.geomgraph.PlanarGraph.call(this);
 25 
 26   this.lineEdgeMap = {};
 27 
 28   this.ptLocator = new jsts.algorithm.PointLocator();
 29 
 30   this.argIndex = argIndex;
 31   this.parentGeom = parentGeom;
 32   this.boundaryNodeRule = boundaryNodeRule ||
 33       jsts.algorithm.BoundaryNodeRule.OGC_SFS_BOUNDARY_RULE;
 34   if (parentGeom != null) {
 35     this.add(parentGeom);
 36   }
 37 };
 38 
 39 jsts.geomgraph.GeometryGraph.prototype = new jsts.geomgraph.PlanarGraph();

 42 /**
 43  * @param {BoundaryNodeRule}
 44  *          boundaryNodeRule
 45  * @param {int}
 46  *          boundaryCount
 47  * @return {int}
 48  */
 49 jsts.geomgraph.GeometryGraph.determineBoundary = function(boundaryNodeRule,
 50     boundaryCount) {
 51   return boundaryNodeRule.isInBoundary(boundaryCount) ? jsts.geom.Location.BOUNDARY
 52       : jsts.geom.Location.INTERIOR;
 53 };

165 /**
166  * @param {Geometry}
167  *          g
168  */
169 jsts.geomgraph.GeometryGraph.prototype.add = function(g) {
170   if (g.isEmpty()) {
171     return;
172   }
173 
174   // check if this Geometry should obey the Boundary Determination Rule
175   // all collections except MultiPolygons obey the rule
176   if (g instanceof jsts.geom.MultiPolygon)
177     this.useBoundaryDeterminationRule = false;
178 
179   if (g instanceof jsts.geom.Polygon)
180     this.addPolygon(g);
181   // LineString also handles LinearRings
182   else if (g instanceof jsts.geom.LineString)
183     this.addLineString(g);
184   else if (g instanceof jsts.geom.Point)
185     this.addPoint(g);
186   else if (g instanceof jsts.geom.MultiPoint)
187     this.addCollection(g);
188   else if (g instanceof jsts.geom.MultiLineString)
189     this.addCollection(g);
190   else if (g instanceof jsts.geom.MultiPolygon)
191     this.addCollection(g);
192   else if (g instanceof jsts.geom.GeometryCollection)
193     this.addCollection(g);
194   else
195     throw new jsts.error.IllegalArgumentError('Geometry type not supported.');
196 };

210 /**
211  * Add an Edge computed externally. The label on the Edge is assumed to be
212  * correct.
213  */
214 jsts.geomgraph.GeometryGraph.prototype.addEdge = function(e) {
215   this.insertEdge(e);
216   var coord = e.getCoordinates();
217   // insert the endpoint as a node, to mark that it is on the boundary
218   this.insertPoint(this.argIndex, coord[0], jsts.geom.Location.BOUNDARY);
219   this.insertPoint(this.argIndex, coord[coord.length - 1],
220       jsts.geom.Location.BOUNDARY);
221 };

233 /**
234  * @param {LineString}
235  *          line
236  * @private
237  */
238 jsts.geomgraph.GeometryGraph.prototype.addLineString = function(line) {
239   var coords = jsts.geom.CoordinateArrays.removeRepeatedPoints(line
240       .getCoordinates());
241 
242   if (coords.length < 2) {
243     this.hasTooFewPoints = true;
244     this.invalidPoint = coords[0];
245     return;
246   }
247 
248   // add the edge for the LineString
249   // line edges do not have locations for their left and right sides
250   var e = new jsts.geomgraph.Edge(coords, new jsts.geomgraph.Label(
251       this.argIndex, jsts.geom.Location.INTERIOR));
252   this.lineEdgeMap[line] = e;
253   this.insertEdge(e);
254   /**
255    * Add the boundary points of the LineString, if any. Even if the LineString
256    * is closed, add both points as if they were endpoints. This allows for the
257    * case that the node already exists and is a boundary point.
258    */
259   if (coords.length >= 2 === false) {
260     throw new jsts.error.IllegalArgumentError(
261         'found LineString with single point');
262   }
263 
264   this.insertBoundaryPoint(this.argIndex, coords[0]);
265   this.insertBoundaryPoint(this.argIndex, coords[coords.length - 1]);
266 };

269 /**
270  * Adds a polygon ring to the graph. Empty rings are ignored.
271  *
272  * The left and right topological location arguments assume that the ring is
273  * oriented CW. If the ring is in the opposite orientation, the left and right
274  * locations must be interchanged.
275  *
276  * @private
277  */
278 jsts.geomgraph.GeometryGraph.prototype.addPolygonRing = function(lr, cwLeft,
279     cwRight) {
280   // don't bother adding empty holes
281   if (lr.isEmpty())
282     return;
283 
284   var coord = jsts.geom.CoordinateArrays.removeRepeatedPoints(lr
285       .getCoordinates());
286 
287   if (coord.length < 4) {
288     this.hasTooFewPoints = true;
289     this.invalidPoint = coord[0];
290     return;
291   }
292 
293   var left = cwLeft;
294   var right = cwRight;
295   if (jsts.algorithm.CGAlgorithms.isCCW(coord)) {
296     left = cwRight;
297     right = cwLeft;
298   }
299   var e = new jsts.geomgraph.Edge(coord, new jsts.geomgraph.Label(
300       this.argIndex, jsts.geom.Location.BOUNDARY, left, right));
301   this.lineEdgeMap[lr] = e;
302 
303   this.insertEdge(e);
304   // insert the endpoint as a node, to mark that it is on the boundary
305   this.insertPoint(this.argIndex, coord[0], jsts.geom.Location.BOUNDARY);
306 };

309 /**
310  * @private
311  */
312 jsts.geomgraph.GeometryGraph.prototype.addPolygon = function(p) {
313   this.addPolygonRing(p.getExteriorRing(), jsts.geom.Location.EXTERIOR,
314       jsts.geom.Location.INTERIOR);
315 
316   for (var i = 0; i < p.getNumInteriorRing(); i++) {
317     var hole = p.getInteriorRingN(i);
318 
319     // Holes are topologically labelled opposite to the shell, since
320     // the interior of the polygon lies on their opposite side
321     // (on the left, if the hole is oriented CW)
322     this.addPolygonRing(hole, jsts.geom.Location.INTERIOR,
323         jsts.geom.Location.EXTERIOR);
324   }
325 };

328 jsts.geomgraph.GeometryGraph.prototype.computeEdgeIntersections = function(g,
329     li, includeProper) {
330   var si = new jsts.geomgraph.index.SegmentIntersector(li, includeProper, true);
331   si.setBoundaryNodes(this.getBoundaryNodes(), g.getBoundaryNodes());
332 
333   var esi = this.createEdgeSetIntersector();
334   esi.computeIntersections(this.edges, g.edges, si);
335 
336   return si;
337 };

340 /**
341  * Compute self-nodes, taking advantage of the Geometry type to minimize the
342  * number of intersection tests. (E.g. rings are not tested for
343  * self-intersection, since they are assumed to be valid).
344  *
345  * @param {LineIntersector}
346  *          li the LineIntersector to use.
347  * @param {boolean}
348  *          computeRingSelfNodes if <false>, intersection checks are optimized
349  *          to not test rings for self-intersection.
350  * @return {SegmentIntersector} the SegmentIntersector used, containing
351  *         information about the intersections found.
352  */
353 jsts.geomgraph.GeometryGraph.prototype.computeSelfNodes = function(li,
354     computeRingSelfNodes) {
355   var si = new jsts.geomgraph.index.SegmentIntersector(li, true, false);
356   var esi = this.createEdgeSetIntersector();
357   // optimized test for Polygons and Rings
358   if (!computeRingSelfNodes &&
359       (this.parentGeom instanceof jsts.geom.LinearRing ||
360           this.parentGeom instanceof jsts.geom.Polygon || this.parentGeom instanceof jsts.geom.MultiPolygon)) {
361     esi.computeIntersections(this.edges, si, false);
362   } else {
363     esi.computeIntersections(this.edges, si, true);
364   }
365   this.addSelfIntersectionNodes(this.argIndex);
366   return si;
367 };

370 /**
371  * @private
372  */
373 jsts.geomgraph.GeometryGraph.prototype.insertPoint = function(argIndex, coord,
374     onLocation) {
375   var n = this.nodes.addNode(coord);
376   var lbl = n.getLabel();
377   if (lbl == null) {
378     n.label = new jsts.geomgraph.Label(argIndex, onLocation);
379   } else
380     lbl.setLocation(argIndex, onLocation);
381 };

384 /**
385  * Adds candidate boundary points using the current {@link BoundaryNodeRule}.
386  * This is used to add the boundary points of dim-1 geometries
387  * (Curves/MultiCurves).
388  */
389 jsts.geomgraph.GeometryGraph.prototype.insertBoundaryPoint = function(argIndex,
390     coord) {
391   var n = this.nodes.addNode(coord);
392   var lbl = n.getLabel();
393   // the new point to insert is on a boundary
394   var boundaryCount = 1;
395   // determine the current location for the point (if any)
396   var loc = jsts.geom.Location.NONE;
397   if (lbl !== null)
398     loc = lbl.getLocation(argIndex, jsts.geomgraph.Position.ON);
399   if (loc === jsts.geom.Location.BOUNDARY)
400     boundaryCount++;
401 
402   // determine the boundary status of the point according to the Boundary
403   // Determination Rule
404   var newLoc = jsts.geomgraph.GeometryGraph.determineBoundary(
405       this.boundaryNodeRule, boundaryCount);
406   lbl.setLocation(argIndex, newLoc);
407 };

410 /**
411  * add edge intersections as self intersections from each edge intersection list
412  * @param argIndex
413  */
414 jsts.geomgraph.GeometryGraph.prototype.addSelfIntersectionNodes = function(
415     argIndex) {
416   this.edges.forEach(function(edge) {
417     edge.eiList.getSortedIntersections().forEach(function(ei) {
418       this.addSelfIntersectionNode(argIndex, ei.coord, edge.getLabel().getLocation(argIndex));
419     }, this);
420   }, this);
421 };

424 /**
425  * Add a node for a self-intersection. If the node is a potential boundary node
426  * (e.g. came from an edge which is a boundary) then insert it as a potential
427  * boundary node. Otherwise, just add it as a regular node.
428  */
429 jsts.geomgraph.GeometryGraph.prototype.addSelfIntersectionNode = function(
430     argIndex, coord, loc) {
431   // if this node is already a boundary node, don't change it
432   if (this.isBoundaryNode(argIndex, coord))
433     return;
434   if (loc === jsts.geom.Location.BOUNDARY && this.useBoundaryDeterminationRule)
435     this.insertBoundaryPoint(argIndex, coord);
436   else
437     this.insertPoint(argIndex, coord, loc);
438 };
