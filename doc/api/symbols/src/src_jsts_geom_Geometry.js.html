  7 /**
  8  * The base class for all geometric objects.
  9  *
 10  *  <H3>Binary Predicates</H3>
 11  * Because it is not clear at this time
 12  * what semantics for spatial
 13  *  analysis methods involving <code>GeometryCollection</code>s would be useful,
 14  *  <code>GeometryCollection</code>s are not supported as arguments to binary
 15  *  predicates (other than <code>convexHull</code>) or the <code>relate</code>
 16  *  method.
 17  *
 18  *  <H3>Set-Theoretic Methods</H3>
 19  *
 20  *  The spatial analysis methods will
 21  *  return the most specific class possible to represent the result. If the
 22  *  result is homogeneous, a <code>Point</code>, <code>LineString</code>, or
 23  *  <code>Polygon</code> will be returned if the result contains a single
 24  *  element; otherwise, a <code>MultiPoint</code>, <code>MultiLineString</code>,
 25  *  or <code>MultiPolygon</code> will be returned. If the result is
 26  *  heterogeneous a <code>GeometryCollection</code> will be returned. <P>
 27  *
 28  *  Because it is not clear at this time what semantics for set-theoretic
 29  *  methods involving <code>GeometryCollection</code>s would be useful,
 30  * <code>GeometryCollections</code>
 31  *  are not supported as arguments to the set-theoretic methods.
 32  *
 33  *  <H4>Representation of Computed Geometries </H4>
 34  *
 35  *  The SFS states that the result
 36  *  of a set-theoretic method is the "point-set" result of the usual
 37  *  set-theoretic definition of the operation (SFS 3.2.21.1). However, there are
 38  *  sometimes many ways of representing a point set as a <code>Geometry</code>.
 39  *  <P>
 40  *
 41  *  The SFS does not specify an unambiguous representation of a given point set
 42  *  returned from a spatial analysis method. One goal of JTS is to make this
 43  *  specification precise and unambiguous. JTS will use a canonical form for
 44  *  <code>Geometry</code>s returned from spatial analysis methods. The canonical
 45  *  form is a <code>Geometry</code> which is simple and noded:
 46  *  <UL>
 47  *    <LI> Simple means that the Geometry returned will be simple according to
 48  *    the JTS definition of <code>isSimple</code>.
 49  *    <LI> Noded applies only to overlays involving <code>LineString</code>s. It
 50  *    means that all intersection points on <code>LineString</code>s will be
 51  *    present as endpoints of <code>LineString</code>s in the result.
 52  *  </UL>
 53  *  This definition implies that non-simple geometries which are arguments to
 54  *  spatial analysis methods must be subjected to a line-dissolve process to
 55  *  ensure that the results are simple.
 56  *
 57  *  <H4> Constructed Points And The Precision Model </H4>
 58  *
 59  *  The results computed by the set-theoretic methods may contain constructed
 60  *  points which are not present in the input <code>Geometry</code>.
 61  *  These new points arise from intersections between line segments in the
 62  *  edges of the input <code>Geometry</code>s. In the general case it is not
 63  *  possible to represent constructed points exactly. This is due to the fact
 64  *  that the coordinates of an intersection point may contain twice as many bits
 65  *  of precision as the coordinates of the input line segments. In order to
 66  *  represent these constructed points explicitly, JTS must truncate them to fit
 67  *  the <code>PrecisionModel</code>.
 68  *
 69  *  Unfortunately, truncating coordinates moves them slightly. Line segments
 70  *  which would not be coincident in the exact result may become coincident in
 71  *  the truncated representation. This in turn leads to "topology collapses" --
 72  *  situations where a computed element has a lower dimension than it would in
 73  *  the exact result.
 74  *
 75  *  When JTS detects topology collapses during the computation of spatial
 76  *  analysis methods, it will throw an exception. If possible the exception will
 77  *  report the location of the collapse.
 78  *
 79  *  #equals(Object) and #hashCode are not overridden, so that when two
 80  *  topologically equal Geometries are added to HashMaps and HashSets, they
 81  *  remain distinct. This behaviour is desired in many cases.
 82  */

359 /**
360  * Computes the centroid of this <code>Geometry</code>. The centroid is equal
361  * to the centroid of the set of component Geometries of highest dimension
362  * (since the lower-dimension geometries contribute zero "weight" to the
363  * centroid)
364  *
365  * @return a {@link Point} which is the centroid of this Geometry.
366  */
367 jsts.geom.Geometry.prototype.getCentroid = function() {
368   if (isEmpty()) {
369     return null;
370   }
371   var cent;
372   var centPt = null;
373   var dim = this.getDimension();
374   if (dim === 0) {
375     cent = new CentroidPoint();
376     cent.add(this);
377     centPt = cent.getCentroid();
378   } else if (dim === 1) {
379     cent = new CentroidLine();
380     cent.add(this);
381     centPt = cent.getCentroid();
382   } else {
383     cent = new CentroidArea();
384     cent.add(this);
385     centPt = cent.getCentroid();
386   }
387   return this.createPointFromInternalCoord(centPt, this);
388 
389 };

523 /**
524  * Tests whether this geometry touches the specified geometry.
525  * <p>
526  * The <code>touches</code> predicate has the following equivalent
527  * definitions:
528  * <ul>
529  * <li>The geometries have at least one point in common, but their interiors do
530  * not intersect.
531  * <li>The DE-9IM Intersection Matrix for the two geometries matches
532  * <code>[FT*******]</code> or <code>[F**T*****]</code> or
533  * <code>[F***T****]</code>
534  * </ul>
535  * If both geometries have dimension 0, this predicate returns
536  * <code>false</code>
537  *
538  * @param {Geometry}
539  *          g the <code>Geometry</code> with which to compare this
540  *          <code>Geometry.</code>
541  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
542  *         touch; Returns <code>false</code> if both <code>Geometry</code>s
543  *         are points.
544  */
545 jsts.geom.Geometry.prototype.touches = function(g) {
546   // short-circuit test
547   if (!this.getEnvelopeInternal().intersects(g.getEnvelopeInternal())) {
548     return false;
549   }
550   return this.relate(g).isTouches(this.getDimension(), g.getDimension());
551 };

554 /**
555  * Tests whether this geometry intersects the specified geometry.
556  * <p>
557  * The <code>intersects</code> predicate has the following equivalent
558  * definitions:
559  * <ul>
560  * <li>The two geometries have at least one point in common
561  * <li>The DE-9IM Intersection Matrix for the two geometries matches
562  * <code>[T********]</code> or <code>[*T*******]</code> or
563  * <code>[***T*****]</code> or <code>[****T****]</code>
564  * <li><code>! g.disjoint(this)</code> (<code>intersects</code> is the
565  * inverse of <code>disjoint</code>)
566  * </ul>
567  *
568  * @param {Geometry}
569  *          g the <code>Geometry</code> with which to compare this
570  *          <code>Geometry.</code>
571  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
572  *         intersect.
573  *
574  * @see Geometry#disjoint
575  */
576 jsts.geom.Geometry.prototype.intersects = function(g) {
577 
578   // short-circuit envelope test
579   if (!this.getEnvelopeInternal().intersects(g.getEnvelopeInternal())) {
580     return false;
581   }
582 
583   /**
584    * TODO: (MD) Add optimizations: - for P-A case: If P is in env(A), test for
585    * point-in-poly - for A-A case: If env(A1).overlaps(env(A2)) test for
586    * overlaps via point-in-poly first (both ways) Possibly optimize selection of
587    * point to test by finding point of A1 closest to centre of env(A2). (Is
588    * there a test where we shouldn't bother - e.g. if env A is much smaller than
589    * env B, maybe there's no point in testing pt(B) in env(A)?
590    */
591 
592   // optimization for rectangle arguments
593   if (this.isRectangle()) {
594     return RectangleIntersects.intersects(this, g);
595   }
596   if (g.isRectangle()) {
597     return RectangleIntersects.intersects(g, this);
598   }
599   // general case
600   return this.relate(g).isIntersects();
601 };

604 /**
605  * Tests whether this geometry crosses the specified geometry.
606  * <p>
607  * The <code>crosses</code> predicate has the following equivalent
608  * definitions:
609  * <ul>
610  * <li>The geometries have some but not all interior points in common.
611  * <li>The DE-9IM Intersection Matrix for the two geometries matches
612  * <ul>
613  * <li><code>[T*T******]</code> (for P/L, P/A, and L/A situations)
614  * <li><code>[T*****T**]</code> (for L/P, A/P, and A/L situations)
615  * <li><code>[0********]</code> (for L/L situations)
616  * </ul>
617  * </ul>
618  * For any other combination of dimensions this predicate returns
619  * <code>false</code>.
620  * <p>
621  * The SFS defined this predicate only for P/L, P/A, L/L, and L/A situations.
622  * JTS extends the definition to apply to L/P, A/P and A/L situations as well,
623  * in order to make the relation symmetric.
624  *
625  * @param {Geometry}
626  *          g the <code>Geometry</code> with which to compare this
627  *          <code>Geometry.</code>
628  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
629  *         cross.
630  */
631 jsts.geom.Geometry.prototype.crosses = function(g) {
632   // short-circuit test
633   if (!this.getEnvelopeInternal().intersects(g.getEnvelopeInternal())) {
634     return false;
635   }
636   return this.relate(g).isCrosses(this.getDimension(), g.getDimension());
637 };

640 /**
641  * Tests whether this geometry is within the specified geometry.
642  * <p>
643  * The <code>within</code> predicate has the following equivalent definitions:
644  * <ul>
645  * <li>Every point of this geometry is a point of the other geometry, and the
646  * interiors of the two geometries have at least one point in common.
647  * <li>The DE-9IM Intersection Matrix for the two geometries matches
648  * <code>[T*F**F***]</code>
649  * <li><code>g.contains(this)</code> (<code>within</code> is the converse

669 /**
670  * Tests whether this geometry contains the specified geometry.
671  * <p>
672  * The <code>contains</code> predicate has the following equivalent
673  * definitions:
674  * <ul>
675  * <li>Every point of the other geometry is a point of this geometry, and the
676  * interiors of the two geometries have at least one point in common.
677  * <li>The DE-9IM Intersection Matrix for the two geometries matches
678  * <code>[T*****FF*]</code>
679  * <li><code>g.within(this)</code> (<code>contains</code> is the converse
680  * of <code>within</code>)
681  * </ul>
682  * An implication of the definition is that "Geometries do not contain their
683  * boundary". In other words, if a geometry A is a subset of the points in the
684  * boundary of a geometry B, <code>B.contains(A) = false</code>
685  *
686  * @param {Geometry}
687  *          g the <code>Geometry</code> with which to compare this
688  *          <code>Geometry.</code>
689  * @return {boolean} <code>true</code> if this <code>Geometry</code>
690  *         contains <code>g.</code>
691  *
692  * @see Geometry#within
693  */
694 jsts.geom.Geometry.prototype.contains = function(g) {
695   // short-circuit test
696   if (!this.getEnvelopeInternal().contains(g.getEnvelopeInternal())) {
697     return false;
698   }
699   // optimization for rectangle arguments
700   if (this.isRectangle()) {
701     return RectangleContains.contains(this, g);
702   }
703   // general case
704   return this.relate(g).isContains();
705 };

710  * <p>
711  * The <code>overlaps</code> predicate has the following equivalent
712  * definitions:
713  * <ul>
714  * <li>The geometries have at least one point each not shared by the other (or
715  * equivalently neither covers the other), they have the same dimension, and the
716  * intersection of the interiors of the two geometries has the same dimension as

719  * <code>[T*T***T**]</code> (for two points or two surfaces) or

740 /**
741  * Tests whether this geometry covers the specified geometry.
742  * <p>
743  * The <code>covers</code> predicate has the following equivalent definitions:
744  * <ul>
745  * <li>Every point of the other geometry is a point of this geometry.
746  * <li>The DE-9IM Intersection Matrix for the two geometries matches
747  * <code>[T*****FF*]</code> or <code>[*T****FF*]</code> or
748  * <code>[***T**FF*]</code> or <code>[****T*FF*]</code>
749  * <li><code>g.coveredBy(this)</code> (<code>covers</code> is the converse
750  * of <code>coveredBy</code>)
751  * </ul>
752  * If either geometry is empty, the value of this predicate is <tt>false</tt>.
753  * <p>
754  * This predicate is similar to {@link #contains}, but is more inclusive (i.e.
755  * returns <tt>true</tt> for more cases). In particular, unlike
756  * <code>contains</code> it does not distinguish between points in the
757  * boundary and in the interior of geometries. For most situations,
758  * <code>covers</code> should be used in preference to <code>contains</code>.
759  * As an added benefit, <code>covers</code> is more amenable to optimization,
760  * and hence should be more performant.
761  *
762  * @param {Geometry}
763  *          g the <code>Geometry</code> with which to compare this
764  *          <code>Geometry.</code>
765  * @return {boolean} <code>true</code> if this <code>Geometry</code> covers
766  *         <code>g.</code>
767  *
768  * @see Geometry#contains
769  * @see Geometry#coveredBy
770  */
771 jsts.geom.Geometry.prototype.covers = function(g) {
772   // short-circuit test
773   if (!this.getEnvelopeInternal().covers(g.getEnvelopeInternal())) {
774     return false;
775   }
776   // optimization for rectangle arguments
777   if (this.isRectangle()) {
778     // since we have already tested that the test envelope is covered
779     return true;
780   }
781   return this.relate(g).isCovers();
782 };

1462 /**
1463  * Returns the first non-zero result of <code>compareTo</code> encountered as
1464  * the two <code>Collection</code>s are iterated over. If, by the time one of
1465  * the iterations is complete, no non-zero result has been encountered, returns
1466  * 0 if the other iteration is also complete. If <code>b</code> completes
1467  * before <code>a</code>, a positive number is returned; if a before b, a
1468  * negative number.
1469  *
1470  * @param {Collection}
1471  *          a a <code>Collection</code> of <code>Comparable</code>s.
1472  * @param {Collection}
1473  *          b a <code>Collection</code> of <code>Comparable</code>s.
1474  * @return {int} the first non-zero <code>compareTo</code> result, if any;
1475  *         otherwise, zero.
1476  */
1477 jsts.geom.Geometry.prototype.compare = function(a, b) {
1478   var i = a.iterator();
1479   var j = b.iterator();
1480   while (i.hasNext() && j.hasNext()) {
1481     var aElement = i.next();
1482     var bElement = j.next();
1483     var comparison = aElement.compareTo(bElement);
1484     if (comparison !== 0) {
1485       return comparison;
1486     }
1487   }
1488   if (i.hasNext()) {
1489     return 1;
1490   }
1491   if (j.hasNext()) {
1492     return -1;
1493   }
1494   return 0;
1495 };
