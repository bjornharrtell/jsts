 14   /**
 15    * @extends jsts.geom.Geometry
 16    * @constructor
 17    */
 18   jsts.geom.LineString = function(points) {
 19     OpenLayers.Geometry.Curve.prototype.initialize.apply(this, arguments);
 20 
 21     this.geometries = this.components;
 22   };
 23   jsts.geom.LineString.prototype = OpenLayers.Geometry.LineString.prototype;
 24 
 25   for (key in jsts.geom.Geometry.prototype) {
 26     jsts.geom.LineString.prototype[key] = jsts.geom.Geometry.prototype[key];

 28 
 29   /**
 30    * @return {jsts.geom.Coordinate[]} this LineString's internal coordinate
 31    *         array.
 32    */
 33   jsts.geom.LineString.prototype.getCoordinates = function() {
 34     return this.components;
 35   };
 36 
 37 
 38   /**
 39    * @return {jsts.geom.Coordinate} The n'th coordinate of this jsts.geom.LineString.
 40    * @param {int}
 41    *          n index.
 42    */
 43   jsts.geom.LineString.prototype.getCoordinateN = function(n) {
 44     return this.components[n];
 45   };

 47 
 48   /**
 49    * @return {jsts.geom.Coordinate} The first coordinate of this LineString or
 50    *         null if empty.
 51    */
 52   jsts.geom.LineString.prototype.getCoordinate = function() {
 53     if (this.isEmpty()) {
 54       return null;
 55     }
 56     return this.getCoordinateN(0);
 57   };
 58 
 59 
 60   /**
 61    * @return {int} LineStrings are always 1-dimensional.
 62    */
 63   jsts.geom.LineString.prototype.getDimension = function() {
 64     return 1;
 65   };
 66 
 67 
 68   /**
 69    * @return {int} dimension of the boundary of this jsts.geom.LineString.
 70    */
 71   jsts.geom.LineString.prototype.getBoundaryDimension = function() {
 72     if (this.isClosed()) {
 73       return Dimension.FALSE;
 74     }
 75     return 0;
 76   };
 77 
 78 
 79   /**
 80    * @return {Boolean} true if empty.
 81    */
 82   jsts.geom.LineString.prototype.isEmpty = function() {
 83     return this.components.length === 0;
 84   };
 85 
 86   jsts.geom.LineString.prototype.getNumPoints = function() {
 87     return this.components.length;
 88   };
 89 
 90   jsts.geom.LineString.prototype.getPointN = function(n) {
 91     return this.getFactory().createPoint(this.components[n]);
 92   };
 93 
 94 
 95   jsts.geom.LineString.prototype.getStartPoint = function() {
 96     if (this.isEmpty()) {
 97       return null;
 98     }
 99     return this.getPointN(0);
100   };
101 
102   jsts.geom.LineString.prototype.getEndPoint = function() {
103     if (this.isEmpty()) {
104       return null;
105     }
106     return this.getPointN(this.getNumPoints() - 1);
107   };
108 
109 
110   /**
111    * @return {Boolean} true if LineString is Closed.
112    */
113   jsts.geom.LineString.prototype.isClosed = function() {
114     if (this.isEmpty()) {
115       return false;
116     }
117     return this.getCoordinateN(0).equals2D(
118         this.getCoordinateN(this.components.length - 1));
119   };
120 
121 
122   /**
123    * @return {Boolean} true if LineString is a Ring.
124    */
125   jsts.geom.LineString.prototype.isRing = function() {
126     return this.isClosed() && this.isSimple();
127   };

129 
130   /**
131    * @return {String} String representation of LineString type.
132    */
133   jsts.geom.LineString.prototype.getGeometryType = function() {
134     return 'LineString';
135   };

137 
138   /**
139    * Gets the boundary of this geometry. The boundary of a lineal geometry is
140    * always a zero-dimensional geometry (which may be empty).
141    *
142    * @return {Geometry} the boundary geometry.
143    * @see Geometry#getBoundary
144    */
145   jsts.geom.LineString.prototype.getBoundary = function() {
146     return (new jsts.operation.BoundaryOp(this)).getBoundary();
147   };
148 
149 
150   jsts.geom.LineString.prototype.computeEnvelopeInternal = function() {
151     if (this.isEmpty()) {
152       return new jsts.geom.Envelope();
153     }
154 
155     var env = new jsts.geom.Envelope();
156     this.components.forEach(function(component) {
157       env.expandToInclude(component);
158     });
159 
160     return env;
161   };

164   /**
165    * @param {Geometry}
166    *          other Geometry to compare this LineString to.
167    * @param {double}
168    *          tolerance Tolerance.
169    * @return {Boolean} true if equal.
170    */
171   jsts.geom.LineString.prototype.equalsExact = function(other, tolerance) {
172     if (!this.isEquivalentClass(other)) {
173       return false;
174     }
175 
176     if (this.components.length !== other.components.length) {
177       return false;
178     }
179 
180     if (this.isEmpty() && other.isEmpty()) {
181       return true;
182     }
183 
184     return this.components.reduce(function(equal, component, i) {
185       return equal &&
186           jsts.geom.Geometry.prototype.equal(component, other.components[i],
187               tolerance);
188     });
189   };

192   /**
193    * Creates and returns a full copy of this {@link LineString} object.
194    * (including all coordinates contained by it).
195    *
196    * @return {LineString} a clone of this instance.
197    */
198   jsts.geom.LineString.prototype.clone = function() {
199     var points = this.components.map(function(component) {
200       return component.clone();
201     });
202 
203     var clone = new jsts.geom.LineString(points);
204 
205     return clone;
206   };
207 
208   jsts.geom.LineString.prototype.isEquivalentClass = function(other) {
209     return other instanceof jsts.geom.LineString;
210   };
211 
212   jsts.geom.LineString.prototype.compareToSameClass = function(o)
213   {
214     var line = o;
215     // MD - optimized implementation
216     var i = 0;
217     var j = 0;
218     while (i < this.components.length && j < line.components.length) {
219       var comparison = this.components[i].compareTo(line.components[j]);
220       if (comparison !== 0) {
221         return comparison;
222       }
223       i++;
224       j++;
225     }
226     if (i < this.components.length) {
227       return 1;
228     }
229     if (j < line.components.length) {
230       return -1;
231     }
232     return 0;
233   };

235   jsts.geom.LineString.prototype.apply = function(filter) {
236     filter.filter(this);
237   };
