 11 /**
 12  * Initializes this class with the openlayers inheritance mechanism
 13  */
 14 jsts.index.ArrayListVisitor.prototype.initialize = function() {
 15   this.items = [];
 16 };
 17 
 18 
 19 /**
 20  * Visits an item
 21  * @param {Object}
 22  *        item the item to visit.
 23  */
 24 jsts.index.ArrayListVisitor.prototype.visitItem = function(item) {
 25   this.items.push(item);
 26 };
 27 
 28 
 29 /**
 30  * Returns all visited items
 31  * @return {Array}
 32  *         An array with all visited items.
 33  */
 34 jsts.index.ArrayListVisitor.prototype.getItems = function() {
 35   return this.items;
 36 };
