  8 /**
  9  * Algorithms for computing values and predicates
 10  * associated with triangles.
 11  *
 12  * For some algorithms extended-precision
 13  * implementations are provided, which are more robust
 14  * (i.e. they produce correct answers in more cases).
 15  * Also, some more robust formulations of
 16  * some algorithms are provided, which utilize
 17  * normalization to the origin.
 18  *

 23 /**
 24  * Tests if a point is inside the circle defined by the points a, b, c.
 25  * This test uses simple
 26  * double-precision arithmetic, and thus may not be robust.
 27  *
 28  * @param {jsts.geom.Coordinate}
 29  *        a a vertex of the triangle.
 30  * @param {jsts.geom.Coordinate}
 31  *        b a vertex of the triangle.
 32  * @param {jsts.geom.Coordinate}
 33  *        c a vertex of the triangle.
 34  * @param {jsts.geom.Coordinate}
 35  *        p the point to test.
 36  * @return {Boolean}
 37  *         true if this point is inside the circle defined
 38  *         by the points a, b, c.
 39  */
 40 jsts.triangulate.quadedge.TrianglePredicate.isInCircleNonRobust =
 41     function(a, b, c, p) {
 42   var isInCircle = (a.x * a.x + a.y * a.y) *
 43           jsts.triangulate.quadedge.TrianglePredicate.triArea(b, c, p) -
 44           (b.x * b.x + b.y * b.y) * jsts.triangulate.quadedge.TrianglePredicate
 45             .triArea(a, c, p) + (c.x * c.x + c.y * c.y) *
 46       jsts.triangulate.quadedge.TrianglePredicate.triArea(a, b, p) -
 47           (p.x * p.x + p.y * p.y) *
 48           jsts.triangulate.quadedge.TrianglePredicate.triArea(a, b, c) > 0;
 49 
 50   return isInCircle;
 51 };

 54 /**
 55  * Tests if a point is inside the circle defined by the points a, b, c.
 56  * This test uses simple
 57  * double-precision arithmetic, and thus is not 10% robust.
 58  * However, by using normalization to the origin
 59  * it provides improved robustness and increased performance.
 60  * <p>
 61  * Based on code by J.R.Shewchuk.
 62  *
 63  *
 64  * @param {jsts.geom.Coordinate}
 65  *        a a vertex of the triangle.
 66  * @param {jsts.geom.Coordinate}
 67  *        b a vertex of the triangle.
 68  * @param {jsts.geom.Coordinate}
 69  *        c a vertex of the triangle.
 70  * @param {jsts.geom.Coordinate}
 71  *        p the point to test.
 72  * @return {Boolean}
 73  *         true if this point is inside the circle defined by the points
 74  *         a, b, c.
 75  */
 76 jsts.triangulate.quadedge.TrianglePredicate.isInCircleNormalized =
 77     function(a, b, c, p) {
 78   var adx, ady, bdx, bdy, cdx, cdy, abdet, bcdet, cadet, alift, blift,
 79       clift, disc;
 80 
 81   adx = a.x - p.x;
 82   ady = a.y - p.y;
 83   bdx = b.x - p.x;
 84   bdy = b.y - p.y;
 85   cdx = c.x - p.x;
 86   cdy = c.y - p.y;
 87 
 88   abdet = adx * bdy - bdx * ady;
 89   bcdet = bdx * cdy - cdx * bdy;
 90   cadet = cdx * ady - adx * cdy;
 91   alift = adx * adx + ady * ady;
 92   blift = bdx * bdx + bdy * bdy;
 93   clift = cdx * cdx + cdy * cdy;
 94 
 95   disc = alift * bcdet + blift * cadet + clift * abdet;
 96   return disc > 0;
 97 };

100 /**
101  * Computes twice the area of the oriented triangle (a, b, c), i.e.
102  * the area is positive if the triangle is oriented counterclockwise.
103  *
104  * @param {jsts.geom.Coordinate}
105  *        a a vertex of the triangle.
106  * @param {jsts.geom.Coordinate}
107  *        b a vertex of the triangle.
108  * @param {jsts.geom.Coordinate}
109  *        c a vertex of the triangle.
110  * @return {Number}
111  *         The calculated area.
112  */
113 jsts.triangulate.quadedge.TrianglePredicate.triArea = function(a, b, c) {
114   return (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
115 };

141 /**
142  * Tests if a point is inside the circle defined by the points a, b, c.
143  * The computation uses {@link DD} arithmetic for robustness.
144  *
145  * @param {jsts.geom.Coordinate}
146  *        a a vertex of the triangle.
147  * @param {jsts.geom.Coordinate}
148  *        b a vertex of the triangle.
149  * @param {jsts.geom.Coordinate}
150  *        c a vertex of the triangle.
151  * @param {jsts.geom.Coordinate}
152  *        p the point to test.
153  * @return {Boolean}
154  *         true if this point is inside the circle defined by the points
155  *         a, b, c.
156  */
157 jsts.triangulate.quadedge.TrianglePredicate.isInCircleDDSlow =
158     function(a, b, c, p) {
159   var px, py, ax, ay, bx, by, cx, cy, aTerm, bTerm, cTerm, pTerm, sum,
160       isInCircle;
161 
162   px = jsts.math.DD.valueOf(p.x);
163   py = jsts.math.DD.valueOf(p.y);
164   ax = jsts.math.DD.valueOf(a.x);
165   ay = jsts.math.DD.valueOf(a.y);
166   bx = jsts.math.DD.valueOf(b.x);
167   by = jsts.math.DD.valueOf(b.y);
168   cx = jsts.math.DD.valueOf(c.x);
169   cy = jsts.math.DD.valueOf(c.y);
170 
171   aTerm = (ax.multiply(ax).add(ay.multiply(ay)))
172     .multiply(jsts.triangulate.quadedge.TrianglePredicate
173         .triAreaDDSlow(bx, by, cx, cy, px, py));
174   bTerm = (bx.multiply(bx).add(by.multiply(by)))
175     .multiply(jsts.triangulate.quadedge.TrianglePredicate
176         .triAreaDDSlow(ax, ay, cx, cy, px, py));
177   cTerm = (cx.multiply(cx).add(cy.multiply(cy)))
178     .multiply(jsts.triangulate.quadedge.TrianglePredicate
179         .triAreaDDSlow(ax, ay, bx, by, px, py));
180   pTerm = (px.multiply(px).add(py.multiply(py)))
181     .multiply(jsts.triangulate.quadedge.TrianglePredicate
182         .triAreaDDSlow(ax, ay, bx, by, cx, cy));
183 
184   sum = aTerm.subtract(bTerm).add(cTerm).subtract(pTerm);
185   isInCircle = sum.doubleValue() > 0;
186 
187   return isInCircle;
188 };

191 /**
192  * Computes twice the area of the oriented triangle (a, b, c), i.e., the area
193  * is positive if the triangle is oriented counterclockwise.
194  * The computation uses {@link DD} arithmetic for robustness.
195  *
196  * @param {jsts.math.DD}
197  *        ax the x ordinate of a vertex of the triangle.
198  * @param {jsts.math.DD}
199  *        ay the y ordinate of a vertex of the triangle.
200  * @param {jsts.math.DD}
201  *        bx the x ordinate of a vertex of the triangle.
202  * @param {jsts.math.DD}
203  *        by the y ordinate of a vertex of the triangle.
204  * @param {jsts.math.DD}
205  *        cx the x ordinate of a vertex of the triangle.
206  * @param {jsts.math.DD}
207  *        cy the y ordinate of a vertex of the triangle.
208  * @return {jsts.math.DD}
209  *         The calculated area.
210  */
211 jsts.triangulate.quadedge.TrianglePredicate.triAreaDDSlow =
212     function(ax, ay, bx, by, cx, cy) {
213   return (bx.subtract(ax).multiply(cy.subtract(ay)).subtract(by.subtract(ay)
214       .multiply(cx.subtract(ax))));
215 };

218 /**
219  * Tests if a point is inside the circle defined by the points a, b, c.
220  * The computation uses {@link DD} arithmetic for robustness.
221  *
222  * @param {jsts.geom.Coordinate}
223  *        a a vertex of the triangle.
224  * @param {jsts.geom.Coordinate}
225  *        b a vertex of the triangle.
226  * @param {jsts.geom.Coordinate}
227  *        c a vertex of the triangle.
228  * @param {jsts.geom.Coordinate}
229  *        p the point to test.
230  * @return {Boolean}
231  *         true if this point is inside the circle defined by the points
232  *         a, b, c.
233  */
234 jsts.triangulate.quadedge.TrianglePredicate.isInCircleDDFast =
235     function(a, b, c, p) {
236   var aTerm, bTerm, cTerm, pTerm, sum, isInCircle;
237 
238   aTerm = (jsts.math.DD.sqr(a.x).selfAdd(jsts.math.DD.
239       sqr(a.y))).selfMultiply(jsts.triangulate.quadedge.TrianglePredicate.
240       triAreaDDFast(b, c, p));
241   bTerm = (jsts.math.DD.sqr(b.x).selfAdd(jsts.math.DD.
242       sqr(b.y))).selfMultiply(jsts.triangulate.quadedge.TrianglePredicate.
243       triAreaDDFast(a, c, p));
244   cTerm = (jsts.math.DD.sqr(c.x).selfAdd(jsts.math.DD.
245       sqr(c.y))).selfMultiply(jsts.triangulate.quadedge.TrianglePredicate.
246       triAreaDDFast(a, b, p));
247   pTerm = (jsts.math.DD.sqr(p.x).selfAdd(jsts.math.DD.
248       sqr(p.y))).selfMultiply(jsts.triangulate.quadedge.TrianglePredicate.
249       triAreaDDFast(a, b, c));
250 
251   sum = aTerm.selfSubtract(bTerm).selfAdd(cTerm).selfSubtract(pTerm);
252   isInCircle = sum.doubleValue() > 0;
253 
254   return isInCircle;
255 };

258 /**
259  * Computes twice the area of the oriented triangle (a, b, c), i.e., the area
260  * is positive if the triangle is oriented counterclockwise.
261  * The computation uses {@link DD} arithmetic for robustness.
262  *
263  * @param {jsts.geom.Coordinate}
264  *        a a vertex in the triangle.
265  * @param {jsts.geom.Coordinate}
266  *        b a vertex in the triangle.
267  * @param {jsts.geom.Coordinate}
268  *        c a vertex in the triangle.
269  * @return {jsts.math.DD}
270  *         The calculated area.
271  */
272 jsts.triangulate.quadedge.TrianglePredicate.triAreaDDFast =
273     function(a, b, c) {
274   var t1, t2;
275 
276   t1 = jsts.math.DD.valueOf(b.x).selfSubtract(a.x).selfMultiply(
277       jsts.math.DD.valueOf(c.y).selfSubtract(a.y));
278 
279   t2 = jsts.math.DD.valueOf(b.y).selSubtract(a.y).selfMultiply(
280       jsts.math.DD.valueOf(c.x).selfSubtract(a.x));
281 
282   return t1.selfSubtract(t2);
283 };

286 /**
287  * Tests if a point is inside the circle defined by the points a, b, c.
288  * This test uses double-double precision arithmetic.
289  *
290  * @param {jsts.geom.Coordinate}
291  *        a a vertex of the triangle.
292  * @param {jsts.geom.Coordinate}
293  *        b a vertex of the triangle.
294  * @param {jsts.geom.Coordinate}
295  *        c a vertex of the triangle.
296  * @param {jsts.geom.Coordinate}
297  *        p the point to test.
298  * @return {Boolean}
299  *         true if this point is inside the circle defined by the
300  *         points a, b, c.
301  */
302 jsts.triangulate.quadedge.TrianglePredicate.
303     isInCircleDDNormalized = function(a, b, c, p) {
304   var adx, ady, bdx, bdy, cdx, cdy, abdet,
305       bcdet, cadet, alift, blift, clift, sum, isInCircle;
306 
307   adx = jsts.math.DD.valueOf(a.x).selfSubtract(p.x);
308   ady = jsts.math.DD.valueOf(a.y).selfSubtract(p.y);
309   bdx = jsts.math.DD.valueOf(b.x).selfSubtract(p.x);
310   bdx = jsts.math.DD.valueOf(b.y).selfSubtract(p.y);
311   cdx = jsts.math.DD.valueOf(c.x).selfSubtract(p.x);
312   cdx = jsts.math.DD.valueOf(c.y).selfSubtract(p.y);
313 
314   abdet = adx.multiply(bdy).selfSubtract(bdx.multiply(ady));
315   bcdet = bdx.multiply(cdy).selfSubtract(cdx.multiply(bdy));
316   cadet = cdx.multiply(ady).selfSubtract(adx.multiply(cdy));
317   alift = adx.multiply(adx).selfAdd(ady.multiply(ady));
318   blift = bdx.multiply(bdx).selfAdd(bdy.multiply(bdy));
319   clift = cdx.multiply(cdx).selfAdd(cdy.multiply(cdy));
320 
321   sum = alift.selfMultiply(bcdet)
322       .selfAdd(blift.selfMultiply(cadet))
323         .selfAdd(clift.selfMultiply(abdet));
324 
325   isInCircle = sum.doubleValue() > 0;
326 
327   return isInCircle;
328 };

331 /**
332  * Computes the inCircle test using distance from the circumcentre.
333  * Uses standard double-precision arithmetic.
334  * <p>
335  * In general this doesn't
336  * appear to be any more robust than the standard calculation. However, there
337  * is at least one case where the test point is far enough from the
338  * circumcircle that this test gives the correct answer.
339  * <pre>
340  * LINESTRING
341  * (1507029.9878 518325.7547, 1507022.1120341457 518332.8225183258,
342  * 1507029.9833 518325.7458, 1507029.9896965567 518325.744909031)
343  * </pre>
344  *
345  * @param {jsts.geom.Coordinate}
346  *        a a vertex of the triangle.
347  * @param {jsts.geom.Coordinate}
348  *        b a vertex of the triangle.
349  * @param {jsts.geom.Coordinate}
350  *        c a vertex of the triangle.
351  * @param {jsts.geom.Coordinate}
352  *        p the point to test.
353  * @return {Boolean}
354  *         true if this point is inside the circle defined by the points
355  *         a, b, c.
356  */
357 jsts.triangulate.quadedge.TrianglePredicate.isInCircleCC =
358     function(a, b, c, p) {
359   var cc, ccRadius, pRadiusDiff;
360 
361   cc = jsts.geom.Triangle.circumcentre(a, b, c);
362   ccRadius = a.distance(cc);
363   pRadiusDiff = p.distance(cc) - ccRadius;
364 
365   return pRadiusDiff <= 0;
366 };
