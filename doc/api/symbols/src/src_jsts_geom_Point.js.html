 18 jsts.geom.Point = function() {

 21 jsts.geom.Point = OpenLayers.Class(jsts.geom.Geometry, {
 22   initialize: function(coordinate) {
 23     if (coordinate === undefined)
 24       return;
 25 
 26     this.coordinate = coordinate;
 27   }
 28 });
 29 
 30 
 31 jsts.geom.Point.prototype.coordinate = null;
 32 
 33 
 34 /**
 35  * @return {double} x-axis value of this Point.
 36  */
 37 jsts.geom.Point.prototype.getX = function() {
 38   return this.coordinate.x;
 39 };
 40 
 41 
 42 /**
 43  * @return {double} y-axis value of this Point.
 44  */
 45 jsts.geom.Point.prototype.getY = function() {
 46   return this.coordinate.y;

 48 
 49 jsts.geom.Point.prototype.getCoordinate = function() {
 50   return this.coordinate;
 51 };
 52 
 53 
 54 /**
 55  * @return {Coordinate[]} this Point as coordinate array.
 56  */
 57 jsts.geom.Point.prototype.getCoordinates = function() {
 58   return this.isEmpty() ? [] : [this.coordinate];
 59 };
 60 
 61 jsts.geom.Point.prototype.isEmpty = function() {
 62   return this.coordinate === null;
 63 };
 64 
 65 jsts.geom.Point.prototype.equalsExact = function(other, tolerance) {
 66   if (!this.isEquivalentClass(other)) {
 67     return false;
 68   }
 69   if (this.isEmpty() && other.isEmpty()) {
 70     return true;
 71   }
 72   return this.equal(other.getCoordinate(), this.getCoordinate(), tolerance);
 73 };

 76 /**
 77  * @return {int} number of coordinates (0 or 1).
 78  */
 79 jsts.geom.Point.prototype.getNumPoints = function() {
 80   return this.isEmpty() ? 0 : 1;
 81 };

 92 /**
 93  * Gets the boundary of this geometry. Zero-dimensional geometries have no
 94  * boundary by definition, so an empty GeometryCollection is returned.
 95  *
 96  * @return {GeometryCollection} an empty GeometryCollection.
 97  * @see Geometry#getBoundary
 98  */
 99 jsts.geom.Point.prototype.getBoundary = function() {
100   return new jsts.geom.GeometryCollection(null);
101 };

104 /**
105  * @return {Envelope} Envelope of this point.
106  */
107 jsts.geom.Point.prototype.computeEnvelopeInternal = function() {
108   if (this.isEmpty()) {
109     return new jsts.geom.Envelope();
110   }
111   return new jsts.geom.Envelope(this);
112 };

114 jsts.geom.Point.prototype.apply = function(filter) {

147 /**
148  * A Point is valid iff:
149  * <ul>
150  * <li>the coordinate which defines it is a valid coordinate (i.e does not have
151  * an NaN X or Y ordinate)
152  * </ul>
153  *
154  * @return {boolean} true iff the Point is valid.
155  */
156 jsts.geom.Point.prototype.isValid = function() {
157   if (!jsts.operation.valid.IsValidOp.isValid(this.getCoordinate())) {
158     return false;
159   }
160   return true;
161 };
