  9 /**
 10  * QuadRoot is the root of a single Quadtree. It is centred at the origin, and
 11  * does not have a defined extent.
 12  *
 13  * @constructor
 14  */
 15 jsts.index.quadtree.Root = OpenLayers.Class(jsts.index.quadtree.NodeBase);
 16 
 17 
 18 /**
 19  * Initilalizes a root using the inheritance mechanism provided by OpenLayers.
 20  */
 21 jsts.index.quadtree.Root.prototype.initialize = function() {
 22   jsts.index.quadtree.NodeBase.prototype.initialize.apply(this, arguments);
 23 
 24   // the root quad is centred at the origin.
 25   this.origin = new jsts.geom.Coordinate(0.0, 0.0);
 26 };
 27 
 28 
 29 /**
 30  * Insert an item into the quadtree this is the root of.
 31  *
 32  * @param {jsts.geom.Envelope}
 33  *          itemEnv the item envelope.
 34  * @param {Object}
 35  *          item the item to insert.
 36  */
 37 jsts.index.quadtree.Root.prototype.insert = function(itemEnv, item) {
 38   var index = this.getSubnodeIndex(itemEnv, this.origin);
 39 
 40   // if index is -1, itemEnv must cross the X or Y axis.
 41   if (index === -1) {
 42     this.add(item);
 43     return;
 44   }
 45   /**
 46    * the item must be contained in one quadrant, so insert it into the tree for
 47    * that quadrant (which may not yet exist)
 48    */
 49   var node = this.subnode[index];
 50   /**
 51    * If the subquad doesn't exist or this item is not contained in it, have to
 52    * expand the tree upward to contain the item.
 53    */
 54 
 55   if (node === null || !node.getEnvelope().contains(itemEnv)) {
 56     var largerNode = jsts.index.quadtree.Node.createExpanded(node, itemEnv);
 57     this.subnode[index] = largerNode;
 58   }
 59   /**
 60    * At this point we have a subquad which exists and must contain contains the
 61    * env for the item. Insert the item into the tree.
 62    */
 63   this.insertContained(this.subnode[index], itemEnv, item);
 64 };
 65 
 66 
 67 /**
 68  * insert an item which is known to be contained in the tree rooted at the given
 69  * QuadNode root. Lower levels of the tree will be created if necessary to hold
 70  * the item.
 71  *
 72  * @param {jsts.index.quadtree.Node}
 73  *          tree the root-node of the tree.
 74  * @param {jsts.geom.Envelope}
 75  *          itemEnv the envelope.
 76  * @param {Object}
 77  *          item the item to insert.
 78  */
 79 jsts.index.quadtree.Root.prototype.insertContained = function(tree, itemEnv,
 80     item) {
 81   /**
 82    * Do NOT create a new quad for zero-area envelopes - this would lead to
 83    * infinite recursion. Instead, use a heuristic of simply returning the
 84    * smallest existing quad containing the query
 85    */
 86   var isZeroX, isZeroY, node;
 87   isZeroX = jsts.index.IntervalSize.isZeroWidth(itemEnv.getMinX(),
 88       itemEnv.getMaxX());
 89   isZeroY = jsts.index.IntervalSize.isZeroWidth(itemEnv.getMinY(),
 90       itemEnv.getMaxY());
 91 
 92   if (isZeroX || isZeroY) {
 93     node = tree.find(itemEnv);
 94   } else {
 95     node = tree.getNode(itemEnv);
 96   }
 97   node.add(item);
 98 };
