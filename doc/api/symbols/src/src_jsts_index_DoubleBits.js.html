  9 /**
 10  * DoubleBits manipulates Double numbers by using bit manipulation and bit-field
 11  * extraction. For some operations (such as determining the exponent) this is
 12  * more accurate than using mathematical operations (which suffer from round-off
 13  * error).
 14  * <p>
 15  * The algorithms and constants in this class apply only to IEEE-754
 16  * double-precision floating point format.
 17  *
 18  * NOTE: Since the only numberformat in JavaScript is IEEE-754 the code in
 19  * DoubleBits could not be easily ported.
 20  *
 21  * Instead, using algorithms found here:
 22  * http://www.merlyn.demon.co.uk/js-exact.htm
 23  *
 24  * @constructor
 25  */
 26 jsts.index.DoubleBits = OpenLayers.Class();
 27 
 28 
 29 /**
 30  * Calculates the power of two for a number
 31  *
 32  * @param {Number}
 33  *          exp value to pow.
 34  * @return {Number} the pow'ed value.
 35  */
 36 jsts.index.DoubleBits.prototype.powerOf2 = function(exp) {
 37   // TODO: Make sure the accuracy of this is sufficient (why else would JTS have
 38   // this in DoubleBits?)
 39   return Math.pow(2, exp);
 40 };
 41 
 42 
 43 /**
 44  * Calculates the exponent-part of the bit-pattern for a number
 45  *
 46  * @param {Number}
 47  *          d the IEEE-754-value to calculate the exponent for.
 48  * @return {Number} the exponent part of the bit-mask.
 49  */
 50 jsts.index.DoubleBits.prototype.exponent = function(d) {
 51   return jsts.index.DoubleBits.prototype.CVTFWD(64, d) - 1023;
 52 };
 53 
 54 
 55 /**
 56  * Calculates the exponent of the bit-pattern for a number. Uses code from:
 57  * http://www.merlyn.demon.co.uk/js-exact.htm
 58  *
 59  * @param {Number}
 60  *          NumW 32 or 64 to denote the number of bits.
 61  * @param {Number}
 62  *          Qty the number to calculate the bit pattern for.
 63  * @return {Number} The integer value of the exponent.
 64  */
 65 jsts.index.DoubleBits.prototype.CVTFWD = function(NumW, Qty) {
 66   var Sign, Expo, Mant, Bin, nb01 = ''; // , OutW = NumW/4
 67   var Inf = {
 68     32: {
 69       d: 0x7F,
 70       c: 0x80,
 71       b: 0,
 72       a: 0
 73     },
 74     64: {
 75       d: 0x7FF0,
 76       c: 0,
 77       b: 0,
 78       a: 0
 79     }
 80   };
 81   var ExW = {
 82     32: 8,
 83     64: 11
 84   }[NumW], MtW = NumW - ExW - 1;
 85 
 86   if (!Bin) {
 87     Sign = Qty < 0 || 1 / Qty < 0; // OK for +-0
 88     if (!isFinite(Qty)) {
 89       Bin = Inf[NumW];
 90       if (Sign) {
 91         Bin.d += 1 << (NumW / 4 - 1);
 92       }
 93       Expo = Math.pow(2, ExW) - 1;
 94       Mant = 0;
 95     }
 96   }
 97 
 98   if (!Bin) {
 99     Expo = {
100       32: 127,
101       64: 1023
102     }[NumW];
103     Mant = Math.abs(Qty);
104     while (Mant >= 2) {
105       Expo++;
106       Mant /= 2;
107     }
108     while (Mant < 1 && Expo > 0) {
109       Expo--;
110       Mant *= 2;
111     }
112     if (Expo <= 0) {
113       Mant /= 2;
114       nb01 = 'Zero or Denormal';
115     }
116     if (NumW === 32 && Expo > 254) {
117       nb01 = 'Too big for Single';
118       Bin = {
119         d: Sign ? 0xFF : 0x7F,
120         c: 0x80,
121         b: 0,
122         a: 0
123       };
124       Expo = Math.pow(2, ExW) - 1;
125       Mant = 0;
126     }
127   }
128 
129   return Expo;
130 };
