  5 /**
  6  * A collection of {@link EdgeEnd}s which obey the following invariant:
  7  * They originate at the same node and have the same direction.
  8  *
  9  * @augments {jsts.geomgraph.EdgeEnd}
 10  * @constructor
 11  */
 12 jsts.operation.relate.EdgeEndBundle = function() {
 13   this.edgeEnds = [];
 14 
 15   var e = arguments[0] instanceof jsts.geomgraph.EdgeEnd ? arguments[0] : arguments[1];
 16 
 17   var edge = e.getEdge();
 18   var coord = e.getCoordinate();
 19   var dirCoord = e.getDirectedCoordinate();
 20   var label = new jsts.geomgraph.Label(e.getLabel());
 21 
 22   jsts.geomgraph.EdgeEnd.call(this, edge, coord,
 23       dirCoord, label);
 24 
 25   this.insert(e);
 26 };
 27

 45 jsts.operation.relate.EdgeEndBundle.prototype.insert = function(e) {
 46   // Assert: start point is the same
 47   // Assert: direction is the same
 48   this.edgeEnds.push(e);

 52 /**
 53  * This computes the overall edge label for the set of edges in this
 54  * EdgeStubBundle. It essentially merges the ON and side labels for each edge.
 55  * These labels must be compatible
 56  */
 57 jsts.operation.relate.EdgeEndBundle.prototype.computeLabel = function(
 58     boundaryNodeRule) {
 59   // create the label. If any of the edges belong to areas,
 60   // the label must be an area label
 61   var isArea = false;
 62   for (var i = 0; i < this.edgeEnds.length; i++) {
 63     var e = this.edgeEnds[i];
 64     if (e.getLabel().isArea())
 65       isArea = true;
 66   }
 67   if (isArea)
 68     this.label = new jsts.geomgraph.Label(jsts.geom.Location.NONE, jsts.geom.Location.NONE,
 69         jsts.geom.Location.NONE);
 70   else
 71     this.label = new jsts.geomgraph.Label(jsts.geom.Location.NONE);
 72 
 73   // compute the On label, and the side labels if present
 74   for (var i = 0; i < 2; i++) {
 75     this.computeLabelOn(i, boundaryNodeRule);
 76     if (isArea)
 77       this.computeLabelSides(i);
 78   }
 79 };

 82 /**
 83  * Compute the overall ON location for the list of EdgeStubs. (This is
 84  * essentially equivalent to computing the self-overlay of a single Geometry)
 85  * edgeStubs can be either on the boundary (eg Polygon edge) OR in the interior
 86  * (e.g. segment of a LineString) of their parent Geometry. In addition,
 87  * GeometryCollections use a {@link BoundaryNodeRule} to determine whether a
 88  * segment is on the boundary or not. Finally, in GeometryCollections it can
 89  * occur that an edge is both on the boundary and in the interior (e.g. a
 90  * LineString segment lying on top of a Polygon edge.) In this case the Boundary
 91  * is given precendence. <br>
 92  * These observations result in the following rules for computing the ON
 93  * location:
 94  * <ul>
 95  * <li> if there are an odd number of Bdy edges, the attribute is Bdy
 96  * <li> if there are an even number >= 2 of Bdy edges, the attribute is Int
 97  * <li> if there are any Int edges, the attribute is Int
 98  * <li> otherwise, the attribute is NULL.
 99  * </ul>
100  *
101  * @private
102  */
103 jsts.operation.relate.EdgeEndBundle.prototype.computeLabelOn = function(
104     geomIndex, boundaryNodeRule) {
105   // compute the ON location value
106   var boundaryCount = 0;
107   var foundInterior = false;
108 
109   for (var i = 0; i < this.edgeEnds.length; i++) {
110     var e = this.edgeEnds[i];
111     var loc = e.getLabel().getLocation(geomIndex);
112     if (loc == jsts.geom.Location.BOUNDARY)
113       boundaryCount++;
114     if (loc == jsts.geom.Location.INTERIOR)
115       foundInterior = true;
116   }
117   var loc = jsts.geom.Location.NONE;
118   if (foundInterior)
119     loc = jsts.geom.Location.INTERIOR;
120   if (boundaryCount > 0) {
121     loc = jsts.geomgraph.GeometryGraph.determineBoundary(boundaryNodeRule,
122         boundaryCount);
123   }
124   this.label.setLocation(geomIndex, loc);
125 
126 };

129 /**
130  * Compute the labelling for each side
131  *
132  * @private
133  */
134 jsts.operation.relate.EdgeEndBundle.prototype.computeLabelSides = function(
135     geomIndex) {
136   this.computeLabelSide(geomIndex, jsts.geomgraph.Position.LEFT);
137   this.computeLabelSide(geomIndex, jsts.geomgraph.Position.RIGHT);
138 };

141 /**
142  * To compute the summary label for a side, the algorithm is: FOR all edges IF
143  * any edge's location is INTERIOR for the side, side location = INTERIOR ELSE
144  * IF there is at least one EXTERIOR attribute, side location = EXTERIOR ELSE
145  * side location = NULL <br>
146  * Note that it is possible for two sides to have apparently contradictory
147  * information i.e. one edge side may indicate that it is in the interior of a
148  * geometry, while another edge side may indicate the exterior of the same
149  * geometry. This is not an incompatibility - GeometryCollections may contain
150  * two Polygons that touch along an edge. This is the reason for
151  * Interior-primacy rule above - it results in the summary label having the
152  * Geometry interior on <b>both</b> sides.
153  *
154  * @private
155  */
156 jsts.operation.relate.EdgeEndBundle.prototype.computeLabelSide = function(
157     geomIndex, side) {
158   for (var i = 0; i < this.edgeEnds.length; i++) {
159     var e = this.edgeEnds[i];
160     if (e.getLabel().isArea()) {
161       var loc = e.getLabel().getLocation(geomIndex, side);
162       if (loc === jsts.geom.Location.INTERIOR) {
163         this.label.setLocation(geomIndex, side, jsts.geom.Location.INTERIOR);
164         return;
165       } else if (loc === jsts.geom.Location.EXTERIOR)
166         this.label.setLocation(geomIndex, side, jsts.geom.Location.EXTERIOR);
167     }
168   }
169 };
