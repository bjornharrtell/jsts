  9 /**
 10  * A map of nodes, indexed by the coordinate of the node
 11  *
 12  * @constructor
 13  */
 14 jsts.geomgraph.NodeMap = function(nodeFactory) {
 15   this.nodeMap = {};
 16   this.nodeFact = nodeFactory;
 17 };
 18 
 19 
 20 /**
 21  * NOTE: In In JSTS a JS object replaces TreeMap. Sorting is done when needed.
 22  */
 23 jsts.geomgraph.NodeMap.prototype.nodeMap = null;
 24 
 25 jsts.geomgraph.NodeMap.prototype.nodeFact = null;
 26 
 27 
 28 /**
 29  * This method expects that a node has a coordinate value.
 30  *
 31  * @param {Coordinate/Node}
 32  *          arg
 33  * @return {Node}
 34  */
 35 jsts.geomgraph.NodeMap.prototype.addNode = function(arg) {
 36   var node, coord;
 37 
 38   if (arg instanceof jsts.geom.Coordinate) {
 39     coord = arg;
 40     node = this.nodeMap[coord];
 41     if (node === undefined) {
 42       node = this.nodeFact.createNode(coord);
 43       this.nodeMap[coord] = node;
 44     }
 45     return node;
 46   } else if (arg instanceof jsts.geomgraph.Node) {
 47     var sn = arg;
 48     node = nodeMap[n.getCoordinate()];
 49     if (node === undefined) {
 50       this.nodeMap[n.getCoordinate()] = n;
 51       return n;
 52     }
 53     node.mergeLabel(n);
 54     return node;
 55   }
 56 };

 59 /**
 60  * Adds a node for the start point of this EdgeEnd (if one does not already
 61  * exist in this map). Adds the EdgeEnd to the (possibly new) node.
 62  *
 63  * @param {EdgeEnd}
 64  *          e
 65  */
 66 jsts.geomgraph.NodeMap.prototype.add = function(e) {
 67   var p = e.getCoordinate();
 68   var n = this.addNode(p);
 69   n.add(e);
 70 };

 73 /**
 74  * @param {Coordinate}
 75  *          coord
 76  * @return {Node} the node if found; null otherwise.
 77  */
 78 jsts.geomgraph.NodeMap.prototype.find = function(coord) {
 79   return this.nodeMap[coord];
 80 };

 82 
 83 /**
 84  * @return {Node[]}
 85  */
 86 jsts.geomgraph.NodeMap.prototype.values = function() {
 87   var array = [];
 88   for (var key in this.nodeMap) {
 89     if (this.nodeMap.hasOwnProperty(key)) {
 90       array.push(this.nodeMap[key]);
 91     }
 92   }
 93 
 94   var compare = function(a,b) {
 95     return a.getCoordinate().compareTo(b.getCoordinate());
 96   };
 97   array.sort(compare);
 98 
 99   return array;
100 };

103 /**
104  * @param {int}
105  *          geomIndex
106  * @return {Node[]}
107  */
108 jsts.geomgraph.NodeMap.prototype.getBoundaryNodes = function(geomIndex) {
109   var bdyNodes = [];
110   var i, values = this.values();
111   for (i = 0; i < values.length; i++) {
112     var node = values[i];
113     if (node.getLabel().getLocation(geomIndex) === jsts.geom.Location.BOUNDARY)
114       bdyNodes.push(node);
115   }
116   return bdyNodes;
117 };
