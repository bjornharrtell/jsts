 13 /**
 14  * A EdgeEndStar is an ordered list of EdgeEnds around a node. They are
 15  * maintained in CCW order (starting with the positive x-axis) around the node
 16  * for efficient lookup and topology building.
 17  *
 18  * @constructor
 19  */
 20 jsts.geomgraph.EdgeEndStar = function() {
 21   this.edgeMap = {};
 22   this.edgeList = null;
 23   this.ptInAreaLocation = [jsts.geom.Location.NONE, jsts.geom.Location.NONE];
 24 };

 61 /**
 62  * Insert an EdgeEnd into the map, and clear the edgeList cache, since the list
 63  * of edges has now changed
 64  *
 65  * @protected
 66  */
 67 jsts.geomgraph.EdgeEndStar.prototype.insertEdgeEnd = function(e, obj) {
 68   this.edgeMap[e] = obj;
 69   this.edgeList = null; // edge list has changed - clear the cache

 73 /**
 74  * @return the coordinate for the node this star is based at.
 75  */
 76 jsts.geomgraph.EdgeEndStar.prototype.getCoordinate = function() {
 77   this.getEdges();
 78   if (this.edgeList.length === 0)
 79     return null;
 80   var e = this.edgeList[0];
 81   return e.getCoordinate();
 82 };
 83 jsts.geomgraph.EdgeEndStar.prototype.getDegree = function() {
 84   this.getEdges();
 85   return this.edgeList.length;
 86 };

 89 /**
 90  * Iterator access to the ordered list of edges is optimized by copying the map
 91  * collection to a list. (This assumes that once an iterator is requested, it is
 92  * likely that insertion into the map is complete).
 93  *
 94  * NOTE: jsts does not support iterators
 95  */
 96 jsts.geomgraph.EdgeEndStar.prototype.getEdges = function() {
 97   //if (this.edgeList == null) {
 98   this.edgeList = [];
 99   for (var key in this.edgeMap) {
100     if (this.edgeMap.hasOwnProperty(key)) {
101       this.edgeList.push(this.edgeMap[key]);
102     }
103   }
104 
105   var compare = function(a,b) {
106     return a.compareTo(b);
107   };
108   this.edgeList.sort(compare);
109   //}
110   return this.edgeList;
111 };

113 jsts.geomgraph.EdgeEndStar.prototype.getNextCW = function(ee) {
114   this.getEdges();
115   var i = this.edgeList.indexOf(ee);
116   var iNextCW = i - 1;
117   if (i === 0)
118     iNextCW = this.edgeList.length - 1;
119   return this.edgeList[iNextCW];
120 };

122 jsts.geomgraph.EdgeEndStar.prototype.computeLabelling = function(geomGraph) {
123   this.computeEdgeEndLabels(geomGraph[0].getBoundaryNodeRule());
124   this.propagateSideLabels(0);
125   this.propagateSideLabels(1);
126 
127   /**
128    * If there are edges that still have null labels for a geometry this must be
129    * because there are no area edges for that geometry incident on this node. In
130    * this case, to label the edge for that geometry we must test whether the
131    * edge is in the interior of the geometry. To do this it suffices to
132    * determine whether the node for the edge is in the interior of an area. If
133    * so, the edge has location INTERIOR for the geometry. In all other cases
134    * (e.g. the node is on a line, on a point, or not on the geometry at all) the
135    * edge has the location EXTERIOR for the geometry.
136    * <p>
137    * Note that the edge cannot be on the BOUNDARY of the geometry, since then
138    * there would have been a parallel edge from the Geometry at this node also
139    * labelled BOUNDARY and this edge would have been labelled in the previous
140    * step.
141    * <p>
142    * This code causes a problem when dimensional collapses are present, since it
143    * may try and determine the location of a node where a dimensional collapse
144    * has occurred. The point should be considered to be on the EXTERIOR of the
145    * polygon, but locate() will return INTERIOR, since it is passed the original
146    * Geometry, not the collapsed version.
147    *
148    * If there are incident edges which are Line edges labelled BOUNDARY, then
149    * they must be edges resulting from dimensional collapses. In this case the
150    * other edges can be labelled EXTERIOR for this Geometry.
151    *
152    * MD 8/11/01 - NOT TRUE! The collapsed edges may in fact be in the interior
153    * of the Geometry, which means the other edges should be labelled INTERIOR
154    * for this Geometry. Not sure how solve this... Possibly labelling needs to
155    * be split into several phases: area label propagation, symLabel merging,
156    * then finally null label resolution.
157    */
158   var hasDimensionalCollapseEdge = [false, false];
159   this.getEdges();
160   for (var i = 0; i < this.edgeList.length; i++) {
161     var e = this.edgeList[i];
162     var label = e.getLabel();
163     for (var geomi = 0; geomi < 2; geomi++) {
164       if (label.isLine(geomi) && label.getLocation(geomi) === jsts.geom.Location.BOUNDARY)
165         hasDimensionalCollapseEdge[geomi] = true;
166     }
167   }
168   for (var i = 0; i < this.edgeList.length; i++) {
169     var e = this.edgeList[i];
170     var label = e.getLabel();
171     for (var geomi = 0; geomi < 2; geomi++) {
172       if (label.isAnyNull(geomi)) {
173         var loc = jsts.geom.Location.NONE;
174         if (hasDimensionalCollapseEdge[geomi]) {
175           loc = jsts.geom.Location.EXTERIOR;
176         } else {
177           var p = e.getCoordinate();
178           loc = this.getLocation(geomi, p, geomGraph);
179         }
180         label.setAllLocationsIfNull(geomi, loc);
181       }
182     }
183   }
184 };

187 /**
188  * @private
189  */
190 jsts.geomgraph.EdgeEndStar.prototype.computeEdgeEndLabels = function(
191     boundaryNodeRule) {
192   // Compute edge label for each EdgeEnd
193   this.getEdges();
194   for (var i = 0; i < this.edgeList.length; i++) {
195     var ee = this.edgeList[i];
196     ee.computeLabel(boundaryNodeRule);
197   }
198 };

201 /**
202  * @private
203  */
204 jsts.geomgraph.EdgeEndStar.prototype.getLocation = function(geomIndex, p, geom) {
205   // compute location only on demand
206   if (this.ptInAreaLocation[geomIndex] === jsts.geom.Location.NONE) {
207     this.ptInAreaLocation[geomIndex] = jsts.algorithm.locate.SimplePointInAreaLocator.locate(p,
208         geom[geomIndex].getGeometry());
209   }
210   return this.ptInAreaLocation[geomIndex];
211 };

213 jsts.geomgraph.EdgeEndStar.prototype.isAreaLabelsConsistent = function(
214     geomGraph) {
215   this.computeEdgeEndLabels(geomGraph.getBoundaryNodeRule());
216   return this.checkAreaLabelsConsistent(0);
217 };

220 /**
221  * @private
222  */
223 jsts.geomgraph.EdgeEndStar.prototype.checkAreaLabelsConsistent = function(
224     geomIndex) {
225   // Since edges are stored in CCW order around the node,
226   // As we move around the ring we move from the right to the left side of the
227   // edge
228   this.getEdges();
229   // if no edges, trivially consistent
230   if (this.edgeList.length <= 0)
231     return true;
232   // initialize startLoc to location of last L side (if any)
233   var lastEdgeIndex = this.edgeList.edges.length - 1;
234   var startLabel = this.edgeList[lastEdgeIndex].getLabel();
235   var startLoc = startLabel.getLocation(geomIndex, jsts.geomgraph.Position.LEFT);
236 
237   // TODO: Assert.isTrue(startLoc != Location.NONE, 'Found unlabelled area
238   // edge');
239 
240   var currLoc = startLoc;
241   for (var i = 0; i < this.edgeList.length; i++) {
242     var e = this.edgeList[i];
243     var label = e.getLabel();
244     // we assume that we are only checking a area
245     // TODO: Assert.isTrue(label.isArea(geomIndex), 'Found non-area edge');
246     var leftLoc = label.getLocation(geomIndex, jsts.geomgraph.Position.LEFT);
247     var rightLoc = label.getLocation(geomIndex, jsts.geomgraph.Position.RIGHT);
248     // check that edge is really a boundary between inside and outside!
249     if (leftLoc === rightLoc) {
250       return false;
251     }
252     // check side location conflict
253     // Assert.isTrue(rightLoc == currLoc, "side location conflict " + locStr);
254     if (rightLoc !== currLoc) {
255       return false;
256     }
257     currLoc = leftLoc;
258   }
259   return true;
260 };

263 /**
264  * @private
265  */
266 jsts.geomgraph.EdgeEndStar.prototype.propagateSideLabels = function(geomIndex) {
267   // Since edges are stored in CCW order around the node,
268   // As we move around the ring we move from the right to the left side of the
269   // edge
270   var startLoc = jsts.geom.Location.NONE;
271 
272   // initialize loc to location of last L side (if any)
273   // System.out.println("finding start location");
274   this.getEdges();
275   for (var i = 0; i < this.edgeList.length; i++) {
276     var e = this.edgeList[i];
277     var label = e.getLabel();
278     if (label.isArea(geomIndex) &&
279         label.getLocation(geomIndex, jsts.geomgraph.Position.LEFT) !== jsts.geom.Location.NONE)
280       startLoc = label.getLocation(geomIndex, jsts.geomgraph.Position.LEFT);
281   }
282 
283   // no labelled sides found, so no labels to propagate
284   if (startLoc === jsts.geom.Location.NONE)
285     return;
286 
287   var currLoc = startLoc;
288   this.getEdges();
289   for (var i = 0; i < this.edgeList.length; i++) {
290     var e = this.edgeList[i];
291     var label = e.getLabel();
292     // set null ON values to be in current location
293     if (label.getLocation(geomIndex, jsts.geomgraph.Position.ON) === jsts.geom.Location.NONE)
294       label.setLocation(geomIndex, jsts.geomgraph.Position.ON, currLoc);
295     // set side labels (if any)
296     if (label.isArea(geomIndex)) {
297       var leftLoc = label.getLocation(geomIndex, jsts.geomgraph.Position.LEFT);
298       var rightLoc = label
299           .getLocation(geomIndex, jsts.geomgraph.Position.RIGHT);
300       // if there is a right location, that is the next location to propagate
301       if (rightLoc !== jsts.geom.Location.NONE) {
302         if (rightLoc !== currLoc)
303           throw new jsts.error.TopologyError('side location conflict', e
304               .getCoordinate());
305         if (leftLoc === jsts.geom.Location.NONE) {
306           // TODO: Assert.shouldNeverReachHere('found single null side (at ' +
307           // e.getCoordinate() + ')');
308         }
309         currLoc = leftLoc;
310       } else {
311         /**
312          * RHS is null - LHS must be null too. This must be an edge from the
313          * other geometry, which has no location labelling for this geometry.
314          * This edge must lie wholly inside or outside the other geometry (which
315          * is determined by the current location). Assign both sides to be the
316          * current location.
317          */
318         // TODO: Assert.isTrue(label.getLocation(geomIndex, Position.LEFT) ==
319         // Location.NONE, 'found single null side');
320         label.setLocation(geomIndex, jsts.geomgraph.Position.RIGHT, currLoc);
321         label.setLocation(geomIndex, jsts.geomgraph.Position.LEFT, currLoc);
322       }
323     }
324   }
325 };

327 jsts.geomgraph.EdgeEndStar.prototype.findIndex = function(eSearch) {
328   this.getEdges(); // force edgelist to be computed
329   for (var i = 0; i < this.edgeList.length; i++) {
330     var e = this.edgeList[i];
331     if (e === eSearch)
332       return i;
333   }
334   return -1;
335 };
