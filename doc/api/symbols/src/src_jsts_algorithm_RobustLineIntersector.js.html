 22 /**
 23  * @param {Coordinate}
 24  *          p
 25  * @param {Coordinate}
 26  *          p1
 27  * @param {Coordinate}
 28  *          p2
 29  */
 30 jsts.algorithm.RobustLineIntersector.prototype.computeIntersection = function(
 31     p, p1, p2) {
 32   this._isProper = false;
 33   // do between check first, since it is faster than the orientation test
 34   if (jsts.geom.Envelope.intersects(p1, p2, p)) {
 35     if ((jsts.algorithm.CGAlgorithms.orientationIndex(p1, p2, p) == 0) &&
 36         (jsts.algorithm.CGAlgorithms.orientationIndex(p2, p1, p) == 0)) {
 37       this._isProper = true;
 38       if (p.equals(p1) || p.equals(p2)) {
 39         this._isProper = false;
 40       }
 41       this.result = jsts.algorithm.LineIntersector.POINT_INTERSECTION;
 42       return;
 43     }
 44   }
 45   this.result = jsts.algorithm.LineIntersector.NO_INTERSECTION;
 46 };

 49 /**
 50  * @param {Coordinate}
 51  *          p1
 52  * @param {Coordinate}
 53  *          p2
 54  * @param {Coordinate}
 55  *          q1
 56  * @param {Coordinate}
 57  *          q2
 58  * @return {int}
 59  * @protected
 60  */
 61 jsts.algorithm.RobustLineIntersector.prototype.computeIntersect = function(p1,
 62     p2, q1, q2) {
 63   this._isProper = false;
 64 
 65   // first try a fast test to see if the envelopes of the lines intersect
 66   if (!jsts.geom.Envelope.intersects(p1, p2, q1, q2))
 67     return jsts.algorithm.LineIntersector.NO_INTERSECTION;
 68 
 69   // for each endpoint, compute which side of the other segment it lies
 70   // if both endpoints lie on the same side of the other segment,
 71   // the segments do not intersect
 72   var Pq1 = jsts.algorithm.CGAlgorithms.orientationIndex(p1, p2, q1);
 73   var Pq2 = jsts.algorithm.CGAlgorithms.orientationIndex(p1, p2, q2);
 74 
 75   if ((Pq1 > 0 && Pq2 > 0) || (Pq1 < 0 && Pq2 < 0)) {
 76     return jsts.algorithm.LineIntersector.NO_INTERSECTION;
 77   }
 78 
 79   var Qp1 = jsts.algorithm.CGAlgorithms.orientationIndex(q1, q2, p1);
 80   var Qp2 = jsts.algorithm.CGAlgorithms.orientationIndex(q1, q2, p2);
 81 
 82   if ((Qp1 > 0 && Qp2 > 0) || (Qp1 < 0 && Qp2 < 0)) {
 83     return jsts.algorithm.LineIntersector.NO_INTERSECTION;
 84   }
 85 
 86   var collinear = Pq1 == 0 && Pq2 == 0 && Qp1 == 0 && Qp2 == 0;
 87   if (collinear) {
 88     return this.computeCollinearIntersection(p1, p2, q1, q2);
 89   }
 90 
 91   /**
 92    * At this point we know that there is a single intersection point (since the
 93    * lines are not collinear).
 94    */
 95 
 96   /**
 97    * Check if the intersection is an endpoint. If it is, copy the endpoint as
 98    * the intersection point. Copying the point rather than computing it ensures
 99    * the point has the exact value, which is important for robustness. It is
100    * sufficient to simply check for an endpoint which is on the other line,
101    * since at this point we know that the inputLines must intersect.
102    */
103   if (Pq1 == 0 || Pq2 == 0 || Qp1 == 0 || Qp2 == 0) {
104     this._isProper = false;
105 
106     /**
107      * Check for two equal endpoints. This is done explicitly rather than by the
108      * orientation tests below in order to improve robustness.
109      *
110      * [An example where the orientation tests fail to be consistent is the
111      * following (where the true intersection is at the shared endpoint POINT
112      * (19.850257749638203 46.29709338043669)
113      *
114      * LINESTRING ( 19.850257749638203 46.29709338043669, 20.31970698357233
115      * 46.76654261437082 ) and LINESTRING ( -48.51001596420236
116      * -22.063180333403878, 19.850257749638203 46.29709338043669 )
117      *
118      * which used to produce the INCORRECT result: (20.31970698357233,
119      * 46.76654261437082, NaN)
120      *
121      */
122     if (p1.equals2D(q1) || p1.equals2D(q2)) {
123       this.intPt[0] = p1;
124     } else if (p2.equals2D(q1) || p2.equals2D(q2)) {
125       this.intPt[0] = p2;
126     }
127 
128     /**
129      * Now check to see if any endpoint lies on the interior of the other
130      * segment.
131      */
132     else if (Pq1 == 0) {
133       this.intPt[0] = new jsts.geom.Coordinate(q1);
134     } else if (Pq2 == 0) {
135       this.intPt[0] = new jsts.geom.Coordinate(q2);
136     } else if (Qp1 == 0) {
137       this.intPt[0] = new jsts.geom.Coordinate(p1);
138     } else if (Qp2 == 0) {
139       this.intPt[0] = new jsts.geom.Coordinate(p2);
140     }
141   } else {
142     this._isProper = true;
143     this.intPt[0] = this.intersection(p1, p2, q1, q2);
144   }
145   return jsts.algorithm.LineIntersector.POINT_INTERSECTION;
146 };

149 /**
150  * @param {Coordinate}
151  *          p1
152  * @param {Coordinate}
153  *          p2
154  * @param {Coordinate}
155  *          q1
156  * @param {Coordinate}
157  *          q2
158  * @return {int}
159  * @private
160  */
161 jsts.algorithm.RobustLineIntersector.prototype.computeCollinearIntersection = function(
162     p1, p2, q1, q2) {
163   var p1q1p2 = jsts.geom.Envelope.intersects(p1, p2, q1);
164   var p1q2p2 = jsts.geom.Envelope.intersects(p1, p2, q2);
165   var q1p1q2 = jsts.geom.Envelope.intersects(q1, q2, p1);
166   var q1p2q2 = jsts.geom.Envelope.intersects(q1, q2, p2);
167 
168   if (p1q1p2 && p1q2p2) {
169     this.intPt[0] = q1;
170     this.intPt[1] = q2;
171     return jsts.algorithm.LineIntersector.COLLINEAR_INTERSECTION;
172   }
173   if (q1p1q2 && q1p2q2) {
174     this.intPt[0] = p1;
175     this.intPt[1] = p2;
176     return jsts.algorithm.LineIntersector.COLLINEAR_INTERSECTION;
177   }
178   if (p1q1p2 && q1p1q2) {
179     this.intPt[0] = q1;
180     this.intPt[1] = p1;
181     return q1.equals(p1) && !p1q2p2 && !q1p2q2 ? jsts.algorithm.LineIntersector.POINT_INTERSECTION
182         : jsts.algorithm.LineIntersector.COLLINEAR_INTERSECTION;
183   }
184   if (p1q1p2 && q1p2q2) {
185     this.intPt[0] = q1;
186     this.intPt[1] = p2;
187     return q1.equals(p2) && !p1q2p2 && !q1p1q2 ? jsts.algorithm.LineIntersector.POINT_INTERSECTION
188         : jsts.algorithm.LineIntersector.COLLINEAR_INTERSECTION;
189   }
190   if (p1q2p2 && q1p1q2) {
191     this.intPt[0] = q2;
192     this.intPt[1] = p1;
193     return q2.equals(p1) && !p1q1p2 && !q1p2q2 ? jsts.algorithm.LineIntersector.POINT_INTERSECTION
194         : jsts.algorithm.LineIntersector.COLLINEAR_INTERSECTION;
195   }
196   if (p1q2p2 && q1p2q2) {
197     this.intPt[0] = q2;
198     this.intPt[1] = p2;
199     return q2.equals(p2) && !p1q1p2 && !q1p1q2 ? jsts.algorithm.LineIntersector.POINT_INTERSECTION
200         : jsts.algorithm.LineIntersector.COLLINEAR_INTERSECTION;
201   }
202   return jsts.algorithm.LineIntersector.NO_INTERSECTION;
203 };

206 /**
207  * This method computes the actual value of the intersection point. To obtain
208  * the maximum precision from the intersection calculation, the coordinates are
209  * normalized by subtracting the minimum ordinate values (in absolute value).
210  * This has the effect of removing common significant digits from the
211  * calculation to maintain more bits of precision.
212  *
213  * @param {Coordinate}
214  *          p1
215  * @param {Coordinate}
216  *          p2
217  * @param {Coordinate}
218  *          q1
219  * @param {Coordinate}
220  *          q2
221  * @return {Coordinate}
222  * @private
223  */
224 jsts.algorithm.RobustLineIntersector.prototype.intersection = function(p1, p2,
225     q1, q2) {
226   var intPt = this.intersectionWithNormalization(p1, p2, q1, q2);
227 
228   /**
229    * Due to rounding it can happen that the computed intersection is outside the
230    * envelopes of the input segments. Clearly this is inconsistent. This code
231    * checks this condition and forces a more reasonable answer
232    *
233    * MD - May 4 2005 - This is still a problem. Here is a failure case:
234    *
235    * LINESTRING (2089426.5233462777 1180182.3877339689, 2085646.6891757075
236    * 1195618.7333999649) LINESTRING (1889281.8148903656 1997547.0560044837,
237    * 2259977.3672235999 483675.17050843034) int point =
238    * (2097408.2633752143,1144595.8008114607)
239    *
240    * MD - Dec 14 2006 - This does not seem to be a failure case any longer
241    */
242   if (!this.isInSegmentEnvelopes(intPt)) {
243     // System.out.println("Intersection outside segment envelopes: " + intPt);
244     // System.out.println("Segments: " + this);
245     // compute a safer result
246     intPt = CentralEndpointIntersector.getIntersection(p1, p2, q1, q2);
247     // System.out.println("Snapped to " + intPt);
248   }
249 
250   if (precisionModel != null) {
251     precisionModel.makePrecise(intPt);
252   }
253 
254   return intPt;
255 };

258 /**
259  * @param {Coordinate}
260  *          p1
261  * @param {Coordinate}
262  *          p2
263  * @param {Coordinate}
264  *          q1
265  * @param {Coordinate}
266  *          q2
267  * @return {Coordinate}
268  * @private
269  */
270 jsts.algorithm.RobustLineIntersector.prototype.intersectionWithNormalization = function(
271     p1, p2, q1, q2) {
272   var n1 = new jsts.geom.Coordinate(p1);
273   var n2 = new jsts.geom.Coordinate(p2);
274   var n3 = new jsts.geom.Coordinate(q1);
275   var n4 = new jsts.geom.Coordinate(q2);
276   var normPt = new jsts.geom.Coordinate();
277   this.normalizeToEnvCentre(n1, n2, n3, n4, normPt);
278 
279   var intPt = this.safeHCoordinateIntersection(n1, n2, n3, n4);
280 
281   intPt.x += normPt.x;
282   intPt.y += normPt.y;
283 
284   return intPt;
285 };

288 /**
289  * Computes a segment intersection using homogeneous coordinates. Round-off
290  * error can cause the raw computation to fail, (usually due to the segments
291  * being approximately parallel). If this happens, a reasonable approximation is
292  * computed instead.
293  *
294  * @param {Coordinate}
295  *          p1 a segment endpoint.
296  * @param {Coordinate}
297  *          p2 a segment endpoint.
298  * @param {Coordinate}
299  *          q1 a segment endpoint.
300  * @param {Coordinate}
301  *          q2 a segment endpoint.
302  * @return {Coordinate} the computed intersection point.
303  * @private
304  */
305 jsts.algorithm.RobustLineIntersector.prototype.safeHCoordinateIntersection = function(
306     p1, p2, q1, q2) {
307   var intPt = null;
308   try {
309     intPt = jsts.algorithm.HCoordinate.intersection(p1, p2, q1, q2);
310   } catch (e) {
311     if (e instanceof jsts.error.NotRepresentableError) {
312       // System.out.println("Not calculable: " + this);
313       // compute an approximate result
314       intPt = jsts.algorithm.CentralEndpointIntersector.getIntersection(p1, p2,
315           q1, q2);
316       // System.out.println("Snapped to " + intPt);
317     } else {
318       throw e;
319     }
320   }
321 
322   return intPt;
323 };

326 /**
327  * Normalize the supplied coordinates so that their minimum ordinate values lie
328  * at the origin. NOTE: this normalization technique appears to cause large
329  * errors in the position of the intersection point for some cases.
330  *
331  * @param {Coordinate}
332  *          n1
333  * @param {Coordinate}
334  *          n2
335  * @param {Coordinate}
336  *          n3
337  * @param {Coordinate}
338  *          n4
339  * @param {Coordinate}
340  *          normPt
341  */
342 jsts.algorithm.RobustLineIntersector.prototype.normalizeToMinimum = function(
343     n1, n2, n3, n4, normPt) {
344   normPt.x = this.smallestInAbsValue(n1.x, n2.x, n3.x, n4.x);
345   normPt.y = this.smallestInAbsValue(n1.y, n2.y, n3.y, n4.y);
346   n1.x -= normPt.x;
347   n1.y -= normPt.y;
348   n2.x -= normPt.x;
349   n2.y -= normPt.y;
350   n3.x -= normPt.x;
351   n3.y -= normPt.y;
352   n4.x -= normPt.x;
353   n4.y -= normPt.y;
354 };

357 /**
358  * Normalize the supplied coordinates to so that the midpoint of their
359  * intersection envelope lies at the origin.
360  *
361  * @param {Coordinate}
362  *          n00
363  * @param {Coordinate}
364  *          n01
365  * @param {Coordinate}
366  *          n10
367  * @param {Coordinate}
368  *          n11
369  * @param {Coordinate}
370  *          normPt
371  */
372 jsts.algorithm.RobustLineIntersector.prototype.normalizeToEnvCentre = function(
373     n00, n01, n10, n11, normPt) {
374   var minX0 = n00.x < n01.x ? n00.x : n01.x;
375   var minY0 = n00.y < n01.y ? n00.y : n01.y;
376   var maxX0 = n00.x > n01.x ? n00.x : n01.x;
377   var maxY0 = n00.y > n01.y ? n00.y : n01.y;
378 
379   var minX1 = n10.x < n11.x ? n10.x : n11.x;
380   var minY1 = n10.y < n11.y ? n10.y : n11.y;
381   var maxX1 = n10.x > n11.x ? n10.x : n11.x;
382   var maxY1 = n10.y > n11.y ? n10.y : n11.y;
383 
384   var intMinX = minX0 > minX1 ? minX0 : minX1;
385   var intMaxX = maxX0 < maxX1 ? maxX0 : maxX1;
386   var intMinY = minY0 > minY1 ? minY0 : minY1;
387   var intMaxY = maxY0 < maxY1 ? maxY0 : maxY1;
388 
389   var intMidX = (intMinX + intMaxX) / 2.0;
390   var intMidY = (intMinY + intMaxY) / 2.0;
391   normPt.x = intMidX;
392   normPt.y = intMidY;
393 
394   /*
395   // equilavalent code using more modular but slower method
396   Envelope env0 = new Envelope(n00, n01);
397   Envelope env1 = new Envelope(n10, n11);
398   Envelope intEnv = env0.intersection(env1);
399   Coordinate intMidPt = intEnv.centre();
400 
401   normPt.x = intMidPt.x;
402   normPt.y = intMidPt.y;
403   */
404 
405   n00.x -= normPt.x;
406   n00.y -= normPt.y;
407   n01.x -= normPt.x;
408   n01.y -= normPt.y;
409   n10.x -= normPt.x;
410   n10.y -= normPt.y;
411   n11.x -= normPt.x;
412   n11.y -= normPt.y;
413 };

416 /**
417  * @param {double}
418  *          x1
419  * @param {double}
420  *          x2
421  * @param {double}
422  *          x3
423  * @param {double}
424  *          x4
425  * @return {double}
426  */
427 jsts.algorithm.RobustLineIntersector.prototype.smallestInAbsValue = function(
428     x1, x2, x3, x4) {
429   var x = x1;
430   var xabs = Math.abs(x);
431   if (Math.abs(x2) < xabs) {
432     x = x2;
433     xabs = Math.abs(x2);
434   }
435   if (Math.abs(x3) < xabs) {
436     x = x3;
437     xabs = Math.abs(x3);
438   }
439   if (Math.abs(x4) < xabs) {
440     x = x4;
441   }
442   return x;
443 };

446 /**
447  * Test whether a point lies in the envelopes of both input segments. A
448  * correctly computed intersection point should return <code>true</code> for
449  * this test. Since this test is for debugging purposes only, no attempt is made
450  * to optimize the envelope test.
451  *
452  * @param {Coordinate}
453  *          intPt
454  * @return {boolean} <code>true</code> if the input point lies within both
455  *         input segment envelopes.
456  * @private
457  */
458 jsts.algorithm.RobustLineIntersector.prototype.isInSegmentEnvelopes = function(
459     intPt) {
460   var env0 = new jsts.geom.Envelope(this.inputLines[0][0],
461       this.inputLines[0][1]);
462   var env1 = new jsts.geom.Envelope(this.inputLines[1][0],
463       this.inputLines[1][1]);
464   return env0.contains(intPt) && env1.contains(intPt);
465 };
