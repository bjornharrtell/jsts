  9 /**
 10  * A Position indicates the position of a Location relative to a graph component
 11  * (Node, Edge, or Area).
 12  *
 13  * @constructor
 14  */
 15 jsts.geomgraph.Position = function() {
 16 
 17 };
 18 
 19 
 20 /**
 21  * An indicator that a Location is <i>on</i> a GraphComponent
 22  *
 23  * @type {int}
 24  */
 25 jsts.geomgraph.Position.ON = 0;
 26 
 27 
 28 /**
 29  * An indicator that a Location is to the <i>left</i> of a GraphComponent
 30  *
 31  * @type {int}
 32  */
 33 jsts.geomgraph.Position.LEFT = 1;
 34 
 35 
 36 /**
 37  * An indicator that a Location is to the <i>right</i> of a GraphComponent
 38  *
 39  * @type {int}
 40  */
 41 jsts.geomgraph.Position.RIGHT = 2;
 42 
 43 
 44 /**
 45  * Returns LEFT if the position is RIGHT, RIGHT if the position is LEFT, or the
 46  * position otherwise.
 47  *
 48  * @param {int}
 49  *          position
 50  * @return {int}
 51  */
 52 jsts.geomgraph.Position.opposite = function(position) {
 53   if (position === jsts.geomgraph.Position.LEFT) {
 54     return jsts.geomgraph.Position.RIGHT;
 55   }
 56   if (position === jsts.geomgraph.Position.RIGHT) {
 57     return jsts.geomgraph.Position.LEFT;
 58   }
 59   return position;
 60 };
